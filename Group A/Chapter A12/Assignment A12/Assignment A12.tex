    \problem{}
        \begin{enumerate}
            \item Events $A$ and $B$ are such that $\P{A} = 0.4$, $\P{B} = 0.3$ and $\P{A \cup B} = 0.5$.
            \begin{enumerate}
                \item Determine whether $A$ and $B$ are mutually exclusive.
                \item Determine whether $A$ and $B$ are independent.
            \end{enumerate}
            \item In a competition, 2 teams ($A$ and $B$) will play each other in the best of 3 games. That is, the first team to win 2 games will be the winner and the competition will end. In the first game, both teams have equal chances of winning. In subsequent games, the probability of team $A$ winning team $B$ given that team $A$ won in the previous game is $p$ and the probability of team $A$ winning team $B$ given that team $A$ lost in the previous game is $\dfrac13$.
            \begin{enumerate}
                \item Illustrate the information with an appropriate tree diagram.
                \item Find the value of $p$ such that team $A$ has equal chances of winning and losing the competition.
            \end{enumerate}
        \end{enumerate}    

    \solution
        \part
            \subpart
                \begin{alignat*}{2}
                    &&\P{A \cup B} &= \P{A} + \P{B} - \P{A \cap B}\\
                    \implies&& 0.5 &= 0.4 + 0.3 - \P{A \cap B}\\
                    \implies&& \P{A \cap B} &= 0.2
                \end{alignat*}
                Since $\P{A \cap B} = 0.2 \neq 0$, $A$ and $B$ are not mutually exclusive.

            \subpart
                \[\P{A}{B} = \dfrac{\P{A \cap B}}{\P{B}} = \dfrac{0.2}{0.3} = \dfrac23\]
                Since $\P{A} = 0.4 \neq \dfrac23 = \P{A}{B}$, $A$ and $B$ are not independent.

        \part
            \subpart
                \begin{center}
                    \begin{tikzpicture}
                        \coordinate (A) at (0, 0);
                        \coordinate[label=above:$A$ wins] (B1) at (2, 1.5);
                        \coordinate[label=below:$B$ wins] (B2) at (2, -1.5);
                        \coordinate[label=above:\underline{$A$ wins}] (C1) at (5, 2);
                        \coordinate[label=below:$B$ wins] (C2) at (5, 1);
                        \coordinate[label=above:$A$ wins] (C3) at (5, -1);
                        \coordinate[label=below:\underline{$B$ wins}] (C4) at (5, -2);
                        \coordinate[label=right:\underline{$A$ wins}] (D1) at (8, 2);
                        \coordinate[label=right:\underline{$B$ wins}] (D2) at (8, 1);
                        \coordinate[label=right:\underline{$A$ wins}] (D3) at (8, -1);
                        \coordinate[label=right:\underline{$B$ wins}] (D4) at (8, -2);
        
                        \draw (A) -- (B1);
                        \draw (A) -- (B2);
                        \draw (B1) -- (C1);
                        \draw (B1) -- (C2);
                        \draw (B2) -- (C3);
                        \draw (B2) -- (C4);
                        \draw (C2) -- (D1);
                        \draw (C2) -- (D2);
                        \draw (C3) -- (D3);
                        \draw (C3) -- (D4);
        
                        \node[above left] at ($(A)! 0.5 !(B1)$) {$0.5$};
                        \node[below left] at ($(A)! 0.5 !(B2)$) {$0.5$};
                        \node[above] at ($(B1)! 0.5 !(C1)$) {$p$};
                        \node[below] at ($(B1)! 0.5 !(C2)$) {$1-p$};
                        \node[above] at ($(B2)! 0.5 !(C3)$) {$1/3$};
                        \node[below] at ($(B2)! 0.5 !(C4)$) {$2/3$};
                        \node[above] at ($(C2)! 0.5 !(D1)$) {$1/3$};
                        \node[below] at ($(C2)! 0.5 !(D2)$) {$2/3$};
                        \node[above] at ($(C3)! 0.5 !(D3)$) {$p$};
                        \node[below] at ($(C3)! 0.5 !(D4)$) {$1-p$};
                    \end{tikzpicture}
                \end{center}

            \subpart
                \begin{align*}
                    \P{\text{$A$ wins competition}} &= \bs{\dfrac12 \cdot p} + \bs{\dfrac12 \cdot (1-p) \cdot \dfrac13} + \bs{\dfrac12 \cdot \dfrac13 \cdot p}\\
                    &= \dfrac{p}2 + \dfrac16
                \end{align*}

                Since we wish for $A$ to have equal chances of winning and losing, \[\P{\text{$A$ wins competition}} = \dfrac{p}2 + \dfrac16 = \dfrac12 \implies \boxed{p = \dfrac23}.\]

    \problem{}
        A Personal Identification Number (PIN) consists of 4 digits in order, where each digit ranges from 0 to 9. Susie has difficulty remembering her PIN. She tries to remember her PIN and wrties down what she thinks it is. The probability that the first digit is correct is $0.8$ and the probability that the second digit is correct is $0.86$. The probability that the first two digits are both correct is $0.72$. Find
        \begin{enumerate}
            \item the probability that the second digit is correct given that the first digit is correct,
            \item the probability that the first digit is correct and the second digit is incorrect,
            \item the probability that the second digit is incorrect given that the first digit is incorrect.
        \end{enumerate}

    \solution
        Let $1D$ be the event that the first digit is correct, and $2D$ be the event that the second digit is correct. We have $\P{1D} = 0.8$, $\P{2D} = 0.86$, and $\P{1D \cap 2D} = 0.72$.

        \part
            \[\P{2D}{1D} = \dfrac{\P{2D \cap 1D}}{\P{1D}} = \dfrac{0.72}{0.8} = \boxed{0.9}\]

        \part
            \[\P{1D \cap 2D'} = \P{1D} - \P{1D \cap 2D} = 0.8 - 0.72 = \boxed{0.08}\]
        
        \part
            \begin{align*}
                \P{2D'}{1D'} &= \dfrac{\P{2D' \cap 1D'}}{\P{1D'}}\\
                &= \dfrac{1 - \P{1D \cup 2D}}{1 - \P{1D}}\\
                &= \dfrac{1 - \bs{\P{1D} + \P{2D} - \P{1D \cap 2D}}}{1 - \P{1D}}\\
                &= \dfrac{1 - (0.8 + 0.86 - 0.72)}{1 - 0.8}\\
                &= \boxed{0.3}
            \end{align*}

    \problem{}
        An international tour group consists of the following seventeen people: a pair of twin sisters and their boyfriends, all from Canada; three policewomen from China; a married couple and their two daughters from Singapore, and a large family from Indonesia, consisting of a man, his wife, his parents and his two sons.

        Four people from the group are randomly chosen to play a game. Find the probability that
        \begin{enumerate}
            \item the four people are all of different nationalities,
            \item the four people are all of the same gender,
            \item the four people are all of different nationalities, given that they are all of the same gender.
        \end{enumerate}

    \solution
        \begin{table}[H]
            \centering
            \begin{tabular}{|c|c|c|c|}
            \hline TALLY & Male & Female & SUBTOTAL \\\hline\hline
            Canada & 2 & 2 & 4 \\\hline
            China & 0 & 3 & 3 \\\hline
            Singapore & 1 & 3 & 4 \\\hline
            Indonesia & 4 & 2 & 6 \\\hline
            SUBTOTAL & 7 & 10 & 17\\\hline
            \end{tabular}
        \end{table}

        \part
            \[\P{\text{all different nationalities}} = \dfrac{4}{17} \cdot \dfrac{3}{16} \cdot \dfrac{4}{15} \cdot \dfrac{6}{14} \cdot 4! = \boxed{\dfrac{72}{595}}\]

        \part
            \[\P{\text{all same gender}} = \dfrac{\comb74 + \comb{10}4}{\comb{17}4} = \boxed{\dfrac7{68}}\]
        
        \part
            \[\P{\text{all different nationalities}}{\text{all female}} = \dfrac2{17} \cdot \dfrac{3}{16} \cdot \dfrac{3}{15} \cdot \dfrac2{14} \cdot 4! = \dfrac{9}{595}\]
            Note that $\P{\text{all different nationalities}}{\text{all male}}$ since there are no males from China, whence 
            \begin{align*}
                &\P{\text{all different nationalities}}{\text{all same gender}}\\
                &\hspace{1em}= \dfrac{\P{\text{all different nationalities $\cap$ all same gender}}}{\P{\text{all same gender}}}\\
                &\hspace{1em}= \dfrac{9/595 + 0}{7/68}\\
                &\hspace{1em}= \boxed{\dfrac{36}{245}}
            \end{align*}
