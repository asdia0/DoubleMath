            \begin{alignat*}{2}
                && z^4 &= 8e^{i\frac43 \pi + 2k\pi i}\\

    \problem{}
        Let
        \begin{align*}
            C &= 1 - \binom{2n}{1} \cos \t + \binom{2n}{2} \cos 2\t - \binom{2n}{3} \cos 3\t + \ldots + \cos 2n\t\\
            S &= - \binom{2n}{1} \sin \t + \binom{2n}{2} \sin 2\t - \binom{2n}{3} \sin 3\t + \ldots + \sin 2n\t
        \end{align*}
        where $n$ is a positive integer.

        Show that $C = (-4)^n \cos{n\t} \sin[2n]{\t/2}$, and find the corresponding expression for $S$.

    \solution
        \begin{align*}
            C + iS &= \sum_{k = 0}^{2n} \binom{2n}{k} (-1)^k \cos k\t + \sum_{k = 0}^{2n} \binom{2n}{k} (-1)^k i \sin k\t\\
            &= \sum_{k = 0}^{2n} \binom{2n}{k} (-1)^k (\cos k\t + i\sin k\t)\\
            &= \sum_{k = 0}^{2n} \binom{2n}{k} (-1)^k e^{ik\t}\\
            &= \sum_{k = 0}^{2n} \binom{2n}{k} (-e^{i\t})^k\\
            &= (1 - e^{i\t})^{2n}\\
            &= \bp{e^{i\t/2}}^{2n} \bp{e^{-i\t/2} - e^{i\t/2}}^{2n}\\
            &= \bp{e^{i\t/2}}^{2n} \bp{e^{i\t/2} - e^{-i\t/2}}^{2n}\\
            &= \bp{e^{i\t/2}}^{2n} \bp{2i\sin{\t/2}}^{2n}\\
            &= e^{in\t} 2^{2n} i^{2n} \sin[2n]{\t/2}\\
            &= e^{in\t} 4^n (-1)^n \sin[2n]{\t/2}\\
            &= (\cos n\t + i\sin n\t) (-4)^n \sin[2n]{\t/2}
        \end{align*}

        Comparing real and imaginary parts, we have $C = (-4)^n \cos{n\t} \sin[2n]{\t/2}$ and
        \boxt{$S = (-4)^n \sin{n\t} \sin[2n]{\t/2}$}

    \problem{}
        Given that $z = \cos\t + \i\sin\t$, show that
        \begin{enumerate}
            \item $z - \dfrac1z = 2i\sin\t$,
            \item $z^n + z^{-n} = 2\cos n\t$.
        \end{enumerate}

        Hence, show that \[\sin^6 \t = \dfrac1{32} (10 - 15\cos2\t + 6\cos4\t - \cos6\t)\] Find a similar expression for $\cos^6 \t$, and hence express $\cos^6 \t - \sin^6 \t$ in the form $a\cos2\t + b\cos6\t$.

    \solution
        \part
            \begin{align*}
                z - \dfrac1z &= z - z^{-1}\\
                &= e^{i\t} - e^{-i\t}\\
                &= (\cos \t + i\sin \t) - (\cos{-\t} + i\sin{-\t})\\
                &= (\cos \t + i\sin \t) - (\cos \t - i\sin \t)\\
                &= 2i\sin \t
            \end{align*}

        \part
            \begin{align*}
                z^n + z^{-n} &= \bp{e^{i\t}}^n + \bp{e^{i\t}}^{-n}\\
                &= e^{in\t} + e^{-in\t}\\
                &= (\cos n\t + i\sin n\t) + (\cos{-n\t} + i\sin{n\t})\\
                &= (\cos n\t + i\sin n\t) + (\cos n\t - i\sin n\t)\\
                &= 2\cos n\t
            \end{align*}
        \dash
        \begin{align*}
            \sin^6 \t &= \dfrac1{(2i)^6} (2i \sin \t)^6\\
            &= -\dfrac1{64} (z - z^{-1})^6\\
            &= -\dfrac1{64} \bs{\binom60 z^6 - \binom61 z^4 + \binom62 z^2 - \binom63 z^0 + \binom64 z^{-2} - \binom65 z^{-4} + \binom66 z^{-6}}\\
            &= -\dfrac1{64} \bs{-20 + 15(z^2 + z^{-2}) -6(z^4 + z^{-4}) + (z^6 + z^{-6})}\\
            &= -\dfrac1{64} \bs{-20 + 15(2\cos2\t) -6(2\cos4\t) + 2\cos6\t}\\
            &= \dfrac1{32} \bp{10 - 15\cos2\t + 6\cos4\t - \cos6\t}
        \end{align*}

        \begin{align*}
            \cos^6 \t &= \dfrac1{2^6} (2\cos\t)^6\\
            &= \dfrac1{64} (z + z^{-1})^6\\
            &= \dfrac1{64} \bs{\binom60 z^6 + \binom61 z^4 + \binom62 z^2 + \binom63 z^0 + \binom64 z^{-2} + \binom65 z^{-4} + \binom66 z^{-6}}\\
            &= \dfrac1{64} \bs{20 + 15(z^2 + z^{-2}) + 6(z^4 + z^{-4}) + (z^6 + z^{-6})}\\
            &= \dfrac1{64} \bs{20 + 15(2\cos2\t) + 6(2\cos4\t) + 2\cos6\t}\\
            &= \dfrac1{32} \bp{10 + 15\cos2\t + 6\cos4\t + \cos6\t}
        \end{align*}

        \begin{align*}
            \cos^6 \t - \sin^6 \t &= \bs{\dfrac1{32} \bp{10 + 15\cos2\t + 6\cos4\t + \cos6\t}}\\
            &- \bs{\dfrac1{32} \bp{10 - 15\cos2\t + 6\cos4\t - \cos6\t}}\\
            &= \dfrac1{32} (30 \cos 2\t + 2\cos 6\t)\\
            &= \dfrac{15}{16} \cos 2\t + \dfrac1{16} \cos 6\t
        \end{align*}

        \boxt{$a = \dfrac{15}{16}$, $b = \dfrac1{16}$}
