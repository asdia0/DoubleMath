    \problem{}
        The complex number $w$ is such that $w w\cconj + 2w = 3 + 4i$, where $w\cconj$ is the complex conjugate of $w$. Find $w$ in the form $a + ib$, where $a$ and $b$ are real.

    \solution
        Note $w w\cconj = \bp{\Re w}^2 + \bp{\Im w}^2 \in \R$.

        Taking the imaginary part of the given equation,
        \begin{alignat*}{2}
            &&\Im{ww\cconj + 2w} &= \Im{3 + 4i}\\
            \implies&&2 \Im{w} &= 4\\
            \implies&&\Im{w} &= 2
        \end{alignat*}

        Taking the real part of the given equation,
        \begin{alignat*}{2}
            &&\Re{ww\cconj + 2w} &= \Re{3 + 4i}\\
            \implies&&\bs{\bp{\Re w}^2 + 2^2} + 2\Re w &= 3\\
            \implies&& \bp{\Re w}^2 + 2\Re {w} + 1 &= 0\\
            \implies&& \bp{\Re w + 1}^2 &= 0\\
            \implies&& \Re{w} &= -1
        \end{alignat*}

        Hence, $w = -1 + 2i$.

        \boxt{$w = -1 + 2i$}

    \problem{}
        Express $(3 - i)^2$ in the form $a + ib$.

        Hence, or otherwise, find the roots of the equation $(z+i)^2 = -8+6i$.

    \solution
        \begin{align*}
            (3 - i)^2 &= 3^2 - 2 \cdot 3 \cdot i + (i)^2\\
            &= 9 - 6i - 1\\
            &= 8 - 6i
        \end{align*}
        \boxt{$(3 - i)^2 = 8 - 6i$}
        \begin{alignat*}{2}
            &&(z+i)^2 &= -8 + 6i\\
            \implies&&-(z + i)^2 &= 8 - 6i\\
            \implies&&i^2 (z + i)^2 &= 8 - 6i\\
            \implies&& (iz - 1)^2 &= 8 - 6i\\
            \implies&& iz - 1 &= \pm (3 - i)\\
            \implies&& z &= \dfrac1i (1 \pm (3 - i))\\
            && &= -i(1 \pm (3 - i))\\
            && &= -i \mp (3i + 1)\\
            && &= -1-4i \lor 1 + 2i
        \end{alignat*}

        \boxt{$z = -1-4i \lor 1 + 2i$}

    \problem{}
        \begin{enumerate}
            \item It is given that $z_1 = 1 + \sqrt3 i$. Find the value of $z_1^3$, showing clearly how you obtain your answer.
            \item Given that $1 + \sqrt3 i$ is a root of the equation \[2z^3 + az^2 + bz + 4 = 0\] find the values of the real numbers $a$ and $b$. Hence, solve the above equation.
        \end{enumerate}

    \solution
        \part
            \begin{alignat*}{2}
                &&z_1 &= 1 + \sqrt3 i\\
                && &= 2 \bp{\dfrac12 + \dfrac{\sqrt3}2 i}\\
                && &= 2 \bp{\cos \dfrac\pi3 + i\sin \dfrac\pi3}\\
                && &= 2e^{i\pi/3}\\
                \implies&&z_1^3 &= \bp{2e^{i\pi/3}}^3\\
                && &= 8 e^{i\pi}\\
                && &= 8 \bp{\cos \pi + i\sin\pi}\\
                && &= -8
            \end{alignat*}

            \boxt{$z_1^3 = -8$}

        \part
            Since $1 + \sqrt3i$ is a root of the given equation, we have
            \begin{alignat*}{2}
                &&2(1 + \sqrt3i)^3 + a(1 + \sqrt3i)^2 + b(1 + \sqrt3 i) + 4 &= 0\\
                \implies&&2 \cdot -8 + a(-2 + 2\sqrt3 i) + b(1 + \sqrt3 i) + 4 &0 0\\
                \implies&&(-2a + b) + \sqrt3(2a + b)i &= 12
            \end{alignat*}
            Comparing real and imaginary parts, we obtain $-2a + b = 12$ and $2a + b = 0$, whence $a = -3$ and $b = 6$.
            \boxt{$a = -3$, $b = 6$}

            Since the coefficients of $2z^3 + az^2 + bz + 4$ are all real, the second root is $(1 + \sqrt3i)\cconj = 1 - \sqrt3 i$. Let the third root be $k$. By Vieta's formula,
            \begin{alignat*}{2}
                &&(1 + \sqrt3i)(1 - \sqrt3i)k &= -\dfrac42\\
                \implies&&4k &= -2\\
                \implies&&k &= -\dfrac12
            \end{alignat*}

            \boxt{The roots of the equation are $1 + \sqrt3i$, $1 - \sqrt3i$ and $-\dfrac12$.}
    \problem{}
        The complex number $z$ is such that $az^2 + bz + a = 0$ where $a$ and $b$ are real constants. It is given that $z = z_0$ is a solution to this equation where $\Im{z_0} \neq 0$.

        \begin{enumerate}
            \item Verify that $z = \dfrac1{z_0}$ is the other solution. Hence, show that $\abs{z_0} = 1$.
        \end{enumerate}

        Take $\Im{z_0} = \dfrac12$ for the rest of the question.

        \begin{enumerate}
            \setcounter{enumi}{1}
            \item Find the possible complex numbers for $z_0$.
            \item If $\Re{z_0} > 0$, find $b$ in terms of $a$.
        \end{enumerate}

    \solution
        \part
            \begin{equation*}
                a \bp{\dfrac1{z_0}^2} + b\bp{\dfrac1{z_0}} + a = \dfrac1{z_0}^2 \bp{a + bz_0 + az_0^2} = 0
            \end{equation*}
            Hence, $z = \dfrac1{z_0}$ is a root of the given equation.

            Since $a, b \in \R$, by the conjugate root theorem, $z_0\cconj = \dfrac1{z_0}$.
            \begin{alignat*}{2}
                &&z_0\cconj &= \dfrac1{z_0}\\
                \implies&&z_0 z_0\cconj &= 1\\
                \implies&& \abs{z_0 z_0\cconj} &= 1\\
                \implies&& \abs{z_0} \abs{z_0\cconj} &= 1\\
                \implies&& \abs{z_0} \abs{z_0} &= 1\\
                \implies&& \abs{z_0} &= 1
            \end{alignat*}
            Note that we reject $\abs{z_0} = -1$ since $\abs{z_0} > 0$.

        \part
            Let $z_0 = x + \dfrac12i$.
            \begin{alignat*}{2}
                &&\abs{x + \dfrac12 i} &= 1\\
                \implies&& x^2 + \bp{\dfrac12}^2 &= 1^2\\
                \implies&&x^2 &= \dfrac34\\
                \implies&&x &= \pm\dfrac{\sqrt3}2
            \end{alignat*}
            \boxt{$z_0 = \dfrac{\sqrt3}2 + \dfrac12i \lor -\dfrac{\sqrt3}2 + \dfrac12i$}

        \part
            Since $\Re{z_0} > 0$, we have $z_0 = \dfrac{\sqrt3}2 + \dfrac12i$. By Vieta's formula,
            \begin{alignat*}{2}
                &&z_0 + \dfrac1{z_0} &= -\dfrac{b}{a}\\
                \implies&&z_0 + z_0\cconj &= -\dfrac{b}{a}\\
                \implies&&2 \Re{z_0} &= -\dfrac{b}{a}\\
                \implies&&\sqrt3 &= -\dfrac{b}{a}\\
                \implies&&b &= -\sqrt3 a
            \end{alignat*}
            \boxt{$b = -\sqrt3 a$}
