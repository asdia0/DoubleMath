    \problem{ACJC Prelim 9758/2017/01/Q5}
        The points $O$, $A$ and $B$ are on a plane such that relative to the point $O$, the points $A$ and $B$ have non-parallel position vectors $\vec a$ and $\vec b$ respectively.
    
        \begin{enumerate}
            \item The point $C$ with position vector $c$ is on the plane $OAB$ such that $OC$ bisects the angle $AOB$. Show that $\left(\dfrac{\vec a}{\abs{\vec a}} - \dfrac{\vec b}{\abs{\vec b}}\right) \cdot \vec c = 0$.
            \item The lines $AB$ and $OC$ intersect at $P$. By first verifying that $\overrightarrow{OC}$ is parallel to $\dfrac{\vec a}{\abs{\vec a}} + \dfrac{\vec b}{\abs{\vec b}}$, show that the ratio of $AP:PB = \abs{\vec a} : \abs{\vec b}$.
        \end{enumerate}
        
    \solution
        \part
            Since $OC$ bisects $\angle AOB$,
            \begin{alignat*}{2}
                &&\angle AOC &= \angle COB\\
                \implies&&\cos \angle AOC &= \cos \angle COB\\
                \implies&&\dfrac{\vec a \cdot \vec c}{\abs{\vec a}\abs{\vec c}} &= \dfrac{\vec b \cdot \vec c}{\abs{\vec b}\abs{\vec c}}\\
                \implies&&\left(\dfrac{\vec a}{\abs{\vec a}}\right) \cdot \vec c &= \left(\dfrac{\vec b}{\abs{\vec b}}\right) \cdot \vec c\\
                \implies&&\left(\dfrac{\vec a}{\abs{\vec a}} - \dfrac{\vec b}{\abs{\vec b}}\right) \cdot \vec c &= 0
            \end{alignat*}

        \part
            \begin{center}
                \begin{tikzpicture}

                    \coordinate[label=above:$\hat{\vec a}$] (A) at (3, 4);

                    \coordinate[label=above:$\hat{\vec a} + \hat{\vec b}$] (AB) at (8, 4);

                    \draw[rotate=53] (2.5, 0.2) -- (2.5, -0.2);

                    \draw pic [draw, angle radius=11mm, ""] {angle = B--O--AB};
                    \draw pic [draw, angle radius=13mm, ""] {angle = B--O--AB};
                \end{tikzpicture}
            \end{center}

            Consider the above diagram. Since $\abs{\hat{\vec a}} = \abs{\hat{\vec b}}$, they form a rhombus. Recall that the diagonals of a rhombus bisect opposite angles. Thus, the sum $\hat{\vec a} + \hat{\vec b}$ bisects $\angle AOB$ and is hence parallel to $\overrightarrow{OC}$.

            \begin{center}

            Consider the above diagram. We have $Q$ on $OB$ such that $OA = OQ$. We also have $R$ such that $OA \parallel QR$ and $OA = AR$. From the earlier discussion, $P$ is the intersection of $OR$ and $AB$.

            Now observe that $\triangle OBP$ is similar to $\triangle RAP$. Let $\lambda$ be the scale factor of $\triangle RAP$ with respect to $\triangle OBP$. We hence have
            \begin{alignat*}{2}
                &&\abs{\vec a} = OA = AR = \lambda OB = \lambda \abs{\vec b} &\text{ and } AP = \lambda BP\\
                \implies&&\dfrac{\abs{\vec a}}{\abs{\vec b}} = \lambda &\text{ and } \dfrac{AP}{BP} = \lambda
            \end{alignat*}
            Thus, $\dfrac{\abs{\vec a}}{\abs{\vec b}} = \dfrac{AP}{BP}$, whence $AP:PB = \abs{\vec a} : \abs{\vec b}$.

    \problem{AJC Prelim 9758/2017/01/Q9}
        The position vectors of $A$, $B$ and $C$ with respect to the origin $O$ are $\vec a$, $\vec b$ and $\vec c$ respectively. It is given that $\overrightarrow{AC} = 4\overrightarrow{CB}$ and $\abs{\vec a + \vec b}^2 = \abs{\vec a}^2 + \abs{\vec b}^2$.

        \begin{enumerate}
            \item By considering $(\vec a + \vec b)\cdot(\vec a + \vec b)$, show that $\vec a$ and $\vec b$ are perpendicular.
            \item Find the length of the projection of $\vec c$ on $\vec a$ in terms of $\abs{\vec a}$.
            \item Given that $F$ is the foot of the perpendicular from $C$ to $OA$ and $\vec f$ denotes the position vector $\overrightarrow{OF}$, state the geometrical meaning of $\abs{\vec c \times \vec f}$.
            \item Two points $X$ and $Y$ move along line segments $OA$ and $AB$ respectively such that
            \begin{align*}
                \overrightarrow{OX} &= (\cos 3t)\vec i + (\sin 3t)\vec j + \dfrac12 \vec k\\
                \overrightarrow{OY} &= (\sin t)\vec i + (\cos t)\vec j - 2\vec k
            \end{align*}
            where $t$ is a real parameter, $0 \leq t \leq 2\pi$. By expressing the scalar product of $\overrightarrow{OX}$ and $\overrightarrow{OY}$ in the form of $p\sin(qt) + r$ where $p$, $q$ and $r$ are real values to be determined, find the greatest value of the angle $XOY$.
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                (\vec a + \vec b) \cdot (\vec a + \vec b) &= \vec a \cdot \vec a + \vec a \cdot \vec b + \vec b \cdot \vec a + \vec b \cdot \vec b\\
                &= \abs{\vec a}^2 + 2\vec a \cdot \vec b + \abs{\vec b}^2\\
                &= \abs{\vec a + \vec b}^2 + 2\vec a \cdot \vec b
            \end{align*}
            Since $(\vec a + \vec b) \cdot (\vec a + \vec b) = \abs{\vec a + \vec b}^2$, we have that $\vec a \cdot \vec b = 0$, whence $\vec a$ and $\vec b$ are perpendicular.

        \part
            \begin{center}
                \begin{tikzpicture}
                    \coordinate[label=below:$O$] (O) at (0, 0);
                    \coordinate[label=below:$A$] (A) at (4, 0);
                    \coordinate[label=above:$B$] (B) at (0, 3);
                    \coordinate[label=above right:$C$] (C) at (4/5, 12/5);
                    \coordinate[label=below:$F$] (F) at (4/5,0);

                    \draw (O)--(A);
                    \draw (O)--(B);
                    \draw (O)--(C);
                    \draw (A)--(B);
                    \draw[dotted] (C)--(F);
                \end{tikzpicture}
            \end{center}

            By the Ratio Theorem, $\overrightarrow{OC} = \dfrac15 \vec a + \dfrac45 \vec b$. Since $F$ lies on $OA$, it has the direction vector $\dfrac15 \vec a$. Thus, $OF$, the length of projection of $\vec c$ on $\vec a$, is $\dfrac15 \abs{\vec a}$.

            \boxt{
                The length of projection of $\vec c$ on $\vec a$ is $\dfrac15 \abs{\vec a}$.
            }

        \part
            $\abs{\vec c \times \vec f}$ is the area of a parallelogram defined by $\vec c$ and $\vec f$.

        \part
            We have $\overrightarrow{OX} = \cveciii{\cos 3t}{\sin 3t}{1/2}$ and $\overrightarrow{OY} = \cveciii{\sin t}{\cos t}{-2}$. Hence,
            \begin{align*}
                \overrightarrow{OX} \cdot \overrightarrow{OY} &= \cveciii{\cos 3t}{\sin 3t}{1/2} \cdot \cveciii{\sin t}{\cos t}{-2}\\
                &= \cos 3t \sin t + \sin 3t \cos t - 1\\
                &= \sin 4t - 1
            \end{align*}
            From the geometric definition of the scalar product, we have
            \begin{alignat*}{2}
                &&\overrightarrow{OX} \cdot \overrightarrow{OY} &= \abs{\cveciii{\cos 3t}{\sin 3t}{1/2}} \abs{ \cveciii{\sin t}{\cos t}{-2}} \cos \angle XOY\\
                \implies&&\sin 4t - 1 &= \sqrt{\cos^2 3t + \sin^2 3t + \left(\dfrac12\right)^2} \sqrt{\sin^2 t + \cos^2 t + (-2)^2} \cos \angle XOY\\
                && &= \sqrt{1 + \dfrac14} \sqrt{1 + 4} \cos \angle XOY\\
                && &= \dfrac52 \cos \angle XOY\\
                \implies&&\cos \angle XOY &= \dfrac25(\sin 4t - 1)
            \end{alignat*}
            Observe that $\angle XOY \in [0, \pi)$, where $\cos \angle XOY$ is decreasing. Hence, the maximum value of $\angle XOY$ occurs when $\cos \angle XOY$ is at a minimum. Since the minimum of $\sin 4t$ is $-1$, we have
            \begin{alignat*}{2}
                &&\min \cos \angle XOY &= \dfrac25(-1-1)\\
                \implies&&\max \angle XOY &= \arccos \left(-\dfrac45\right)\\
                && &= 2.50 \tosf{3}                
            \end{alignat*}

            \boxt{
                The greatest value of $\angle XOY$ is $2.50$.
            }

    \problem{CJC Prelim 9758/2017/02/Q2}
        Referred to the origin $O$, the points $A$, $B$, $P$ and $Q$ have position vectors $\vec a$, $\vec b$, $\vec p$ and $\vec q$ respectively, such that $\abs{\vec a} = 2$, $\vec b$ is a unit vector, and the angle between $\vec a$ and $\vec b$ is $\dfrac\pi4$.

        \begin{enumerate}
            \item Give a geometrical interpretation of $\abs{\vec b \cdot \vec a}$.
            \item Find $\abs{\vec a \times \vec b}$, leaving your answer in exact form.
        \end{enumerate}

        \noindent It is also given that $\vec p = 3\vec a + (\mu + 2)\vec b$ and $\vec q = (\mu + 3)\vec a + \mu \vec b$, where $\mu \in \mathbb{R}$.

        \begin{enumerate}
            \setcounter{enumi}{2}
            \item Show that $\vec p \times \vec q = \left(\mu^2 + 2\mu + 6\right)(\vec b \times \vec a)$.
            \item Hence find the smallest area of the triangle $OPQ$ as $\mu$ varies.
        \end{enumerate}

    \solution
        \part
            $\abs{\vec b \cdot \vec a}$ is the area of the parallelogram defined by $\vec b$ and $\vec a$.

        \part
            Let $\theta = \dfrac\pi4$ be the angle between $\vec a$ and $\vec b$.
            \begin{align*}
                \abs{\vec a \times \vec b} &= \abs{\vec a} \abs{\vec b} \sin \theta\\
                &= 2 \cdot 1 \cdot \sin \dfrac\pi4\\
                &= \sqrt2
            \end{align*}

            \boxt{
                $\abs{\vec a \times \vec b} = \sqrt2$
            }

        \part
            \begin{align*}
                \vec p \times \vec q &= \Big[3\vec a + (\mu + 2)\vec b \Big] \times \Big[(\mu + 3)\vec a + \mu \vec b\Big]\\
                &= 3(\mu + 3)\vec a \times \vec a + 3\mu\vec a \times \vec b + (\mu + 2)(\mu + 3)\vec b \times \vec a + \mu(\mu + 2)\vec b \times \vec b\\
                &= 3\mu\vec a \times \vec b + (\mu + 2)(\mu + 3)\vec b \times \vec a\\
                &= -3\mu\vec b \times \vec a + \left(\mu^2 + 5\mu + 6\right)\vec b \times \vec a\\
                &= \left(\mu^2 + 2\mu + 6\right)\vec b \times \vec a
            \end{align*}

        \part
            {\allowdisplaybreaks
            \begin{align*}
                \min \area \triangle OPQ &= \min \dfrac12 \abs{\vec p \times \vec q}\\
                &= \min \dfrac12 \abs{\mu^2 + 2\mu + 6} \abs{\vec b \times \vec a}\\
                &= \min \dfrac12 \abs{(\mu + 1)^2 + 5} \sqrt{2}\\
                &= \dfrac12 \cdot 5 \cdot \sqrt{2}\\
                &= \dfrac{5}{\sqrt{2}}
            \end{align*}
            }

            \boxt{
                The smallest area of $\triangle OPQ$ is $\dfrac5{\sqrt2}$ units$^2$.
            }

    \problem{IJC Prelim 9758/2017/01/Q3}
        The vectors $\vec p$ and $\vec q$ are given by $\vec p = 2\vec i + \vec j + a\vec k$ and $\vec q = b\vec i + \vec j$, where $a$ and $b$ are non-zero constants.

        \begin{enumerate}
            \item Find $(2\vec p - 5\vec q) \times (2\vec p + 5\vec q)$ in terms of $a$ and $b$.
        \end{enumerate}

        \noindent Given that the $\vec i$- and $\vec j-$ components of the answer to part (a) are equal, find the value of $b$. Use the value of $b$ you have found to solve parts (b) and (c).

        \begin{enumerate}
            \setcounter{enumi}{1}
            \item Given that the magnitude of $(2\vec p - 5\vec q) \times (2\vec p + 5\vec q)$ is 80, find the possible exact values of $a$.
            \item Given instead that $2\vec p - 5\vec q$ and $2\vec p + 5\vec q$ are perpendicular, find the exact value of $\abs{\vec p}$.
        \end{enumerate}

    \solution
        \part
            We have $\vec p = \cveciii21a$ and $\vec q = \cveciii{b}10$. Hence,
            \begin{align*}
                (2\vec p - 5\vec q) \times (2\vec p + 5\vec q) &= 4\vec p \times \vec p + 10\vec p \times \vec q - 10\vec q \times \vec p - 25 \vec q \times \vec q\\
                &= 10\vec p \times \vec q - 10\vec q \times \vec p\\
                &= 10\vec p \times \vec q + 10\vec p \times \vec q\\
                &= 20\vec p \times \vec q\\
                &= 20 \cveciii21a \times \cveciii{b}10\\
                &= 20 \cveciii{-a}{ab}{2-b}
            \end{align*}

            \boxt{
                $(2\vec p - 5\vec q) \times (2\vec p + 5\vec q) = 20 \cveciii{-a}{ab}{2-b}$
            }

            Since the $\vec i$- and $\vec j$-components are equal, we have
            \begin{alignat*}{2}
                &&-a &= ab\\
                \implies&&ab + a &= 0\\
                \implies&&a(b+1) &= 0
            \end{alignat*}
            We thus have $b = -1$. Note that we reject $a = 0$ since $a$ is non-zero.

            \boxt{
                $b = -1$
            }

        \part
            \begin{alignat*}{2}
                &&\abs{(2\vec p - 5\vec q) \times (2\vec p + 5\vec q)} &= 80\\
                \implies&&\abs{20 \cveciii{-a}{-a}{3}} &= 80\\
                \implies&&\abs{\cveciii{-a}{-a}{3}} &= 4\\
                \implies&&\sqrt{(-a)^2 + (-a)^2 + 3^2} &= 4\\
                \implies&&2a^2 + 9 &= 16\\
                \implies&&a^2 &= \dfrac72\\
                \implies&&a &= \pm \sqrt{\dfrac72}
            \end{alignat*}

            \boxt{
                $a = \pm \sqrt{\dfrac72}$
            }

        \part
            Since $2\vec p - 5\vec q$ and $2\vec p + 5\vec q$, their dot product is 0.
            \begin{alignat*}{2}
                &&(2\vec p - 5\vec q) \cdot (2\vec p + 5\vec q) &= 0\\
                \implies&&4\vec p \cdot \vec p + 10 \vec p \cdot \vec q - 10 \vec q \cdot \vec p - 25 \vec q \cdot \vec q &= 0\\
                \implies&&4\vec p \cdot \vec p - 25 \vec q \cdot \vec q &= 0\\
                \implies&&4 \abs{\vec p}^2 - 25\abs{\vec q}^2 &= 0\\
                \implies&&4 \abs{\vec p}^2 - 25\abs{\cveciii{-1}10}^2 &= 0\\
                \implies&&4 \abs{\vec p}^2 - 25\cdot2 &= 0\\
                \implies&&\abs{\vec p}^2 &= \dfrac{25}2\\
                \implies&&\abs{\vec p} &= \dfrac5{\sqrt2}
            \end{alignat*}

            Note that we reject $\abs{\vec p} = -\dfrac5{\sqrt2}$ since $\abs{\vec p} \geq 0$.

            \boxt{
                $\abs{\vec p} = \dfrac5{\sqrt2}$
            }

    \problem{JJC Prelim 9758/2017/01/Q6}
        With respect to the origin $O$, the position vectors of the points $U$, $V$ and $W$ are $\vec u$, $\vec v$ and $\vec w$ respectively. The mid-points of the sides $VW$, $WU$ and $UV$ of the triangle $UVW$ are $M$, $N$ and $P$ respectively.

        \begin{enumerate}
            \item Show that $\overrightarrow{UM} = \dfrac12 (\vec v + \vec w - 2\vec u)$.
            \item Find the vector equations of the lines $UM$ and $VN$. Hence show that the position vector of the point of intersection, $G$, of $UM$ and $VN$ is $\dfrac13 (\vec u + \vec v + \vec w)$.
        \end{enumerate}

    \solution
        \part
            By the Midpoint Theorem,
            \begin{alignat*}{2}
                &&\overrightarrow{OM} &= \dfrac12 (\vec v + \vec w)\\
                \implies&&\overrightarrow{OU} + \overrightarrow{UM} &= \dfrac12 (\vec v + \vec w)\\
                \implies&&\overrightarrow{UM} &= \dfrac12 (\vec v + \vec w) - \vec u\\
                && &= \dfrac12 (\vec v + \vec w - 2\vec u)
            \end{alignat*}

        \part
            Note that the line $UM$ has direction vector $\vec v + \vec w - 2\vec u$ and passes through $U$ Hence,
            \boxt{
                $l_{UM} : \vec r = \vec u + \lambda (\vec v + \vec w - 2\vec u), \, \lambda \in \mathbb{R}$
            }
            From the Midpoint Theorem, we have $\overrightarrow{ON} = \dfrac12(\vec w + \vec u)$. Thus, $\overrightarrow{VN} = \overrightarrow{ON} - \overrightarrow{OV} = \dfrac12(\vec w + \vec u - 2\vec v)$. Thus, line $VN$ has direction vector $\vec w + \vec u - 2\vec v$ and passes through $V$.
            \boxt{
                $l_{VN} : \vec r = \vec v + \mu (\vec w + \vec u - 2\vec v), \, \mu \in \mathbb{R}$
            }
            Consider $l_{UM} = l_{VN}$.
            \begin{alignat*}{2}
                &&l_{UM} &= l_{VN}\\
                \implies&&\vec u + \lambda (\vec v + \vec w - 2\vec u) &= \vec v + \mu (\vec w + \vec u - 2\vec v)\\
                \implies&&(1 -2\lambda)\vec u + \lambda \vec v + \lambda \vec w &= \mu \vec u + (1 - 2\mu)\vec v + \mu \vec w
            \end{alignat*}
            Comparing coefficients of $\vec u$, $\vec v$ and $\vec w$ terms, we have the system:
            \begin{equation*}
                \systeme{1-2\lambda=\mu,\lambda=1-2\mu,\lambda=\mu}
            \end{equation*}
            which has solution $\lambda = \mu = \dfrac13$. Thus,
            \begin{align*}
                \overrightarrow{OG} &= \vec v + \dfrac13(\vec w + \vec u - \vec v)\\
                &= \dfrac13(\vec u + \vec v + \vec w)
            \end{align*}
            \boxt{
               $\overrightarrow{OG} = \dfrac13(\vec u + \vec v + \vec w)$ 
            }
            

    \problem{MI Prelim 9740/2017/01/Q5}
        A line $L$ passes through the points $A\coordiii3{-1}0$ and $B\coordiii{11}{11}4$.
        
        \begin{enumerate}
            \item Find the angle between $L$ and the $y$-axis.
            \item State the geometrical meaning of $\abs{\overrightarrow{OB} \cdot \cveciii001}$.
        \end{enumerate}

        \noindent The point $F\coordiii{2a+1}{a}{a-1}$ is a point on $L$, where $a$ is a positive constant. The point $P$ is such that $\overrightarrow{PF} = \cveciii302$ and the area of the triangle $AFP$ is $\sqrt{\dfrac{59}2}$ units$^2$.

        \begin{enumerate}
            \setcounter{enumi}{2}
            \item Determine the value of $a$.
            \item The point $C$ on $L$ is such that the ratio of the area of triangle $AFP$ to the area of triangle $FCP$ is $2:1$. State the ratio $AF:CF$, justifying your answer.
        \end{enumerate}

    \solution
        \part
            Note that $\overrightarrow{AB} = \overrightarrow{OB} - \overrightarrow{OA} = \cveciii{11}{11}4 - \cveciii3{-1}0 = \cveciii8{12}4 = 4\cveciii231$. Since $L$ passes through $A$, it has the vector equation
            \begin{equation*}
                L : \vec r = \cveciii3{-1}0 + \lambda\cveciii231, \, \lambda \in \mathbb{R}
            \end{equation*}
            Observe that the $y$-axis has vector equation $\vec r = \mu\cveciii010$, where $\mu \in \mathbb{R}$. Let $\theta$ be the angle between $L$ and the $y$-axis.
            \begin{alignat*}{2}
                &&\cos\theta &= \dfrac{\cveciii231 \cdot \cveciii010}{\abs{\cveciii231}\abs{\cveciii010}}\\
                && &= \dfrac{3}{\sqrt{14}}\\
                \implies&&\theta &= \arccos \dfrac3{\sqrt14}\\
                && &= 0.641 \tosf{3}
            \end{alignat*}

            \boxt{
                The angle between $L$ and the $y$-axis is $0.641$.
            }

        \part
            $\abs{\overrightarrow{OB} \cdot \cveciii001}$ is the length of projection of $\overrightarrow{OB}$ on the $z$-axis.

        \part
            Since $F$ is on the line $L$, we have that
            \begin{equation*}
                \cveciii{2a+1}{a}{a-1} = \cveciii3{-1}0 + \lambda\cveciii231
            \end{equation*}
            for some $\lambda \in \mathbb{R}$. This gives the system
            \begin{equation*}
                \systeme{2a+1=3+2\lambda, a=-1+3\lambda,a-1=\lambda}
            \end{equation*}
            which has solution $a = 2$, $\lambda = 1$.

            \boxt{
                $a = 2$
            }

        \part
            Since $\triangle AFP$ and $\triangle FCP$ have the same height, the length of the bases of both triangles are in the same ratio as their area. Hence, $AF : CF = \area \triangle AFP : \area \triangle FCP = 2 : 1$.

            \boxt{
                $AF : CF = 2 : 1$
            }

    \problem{NYJC Prelim 9758/2017/02/Q1}
        The position vectors of points $A$ and $B$ with respect to the origin $O$ are $\vec a$ and $\vec b$ respectively where $\vec a$ and $\vec b$ are non-zero vectors. Point $C$ lies on $OA$ produced such that $4OA = AC$ and point $D$ lies on $OB$ produced such that $OB = BD$. The line $BC$ and $AD$ meet at the point $M$.

        \begin{enumerate}
            \item Giving a necessary condition for $\vec a$ and $\vec b$, find the position vector of $M$ in terms of $\vec a$ and $\vec b$.
            \item If $\abs{\vec a} = 1$ and $\abs{\vec b} = 2$, find the shortest distance of $M$ from the line $OC$, giving your answer in the form $k\abs{\vec a \times \vec b}$, where $k$ is a constant to be determined.
        \end{enumerate}

    \solution
        \part
            \boxt{
                $\vec a$ and $\vec b$ must be non-parallel.
            }

            \noindent Note that $\overrightarrow{OC} = 5\vec a$ and $\overrightarrow{OD} = 2\vec b$. Hence, $\overrightarrow{AD} = 2\vec b - \vec a$ and $\overrightarrow{BC} = 5\vec a - \vec b$. Thus, the lines $AD$ and $BC$ have vector equations
            \begin{align*}
                l_{AD} : \vec r &= \vec a + \lambda(2\vec b - \vec a), \, \lambda \in \mathbb{R}\\
                l_{BC} : \vec r &= \vec b + \mu(5\vec a - \vec b), \, \mu \in \mathbb{R}
            \end{align*}
            Consider $l_{AD} = l_{BC}$.
            \begin{alignat*}{2}
                &&l_{AD} &= l_{BC}\\
                \implies&&\vec a + \lambda(2\vec b - \vec a) &= \vec b + \mu(5\vec a - \vec b)
            \end{alignat*}
            Comparing coefficients of $\vec a$ and $\vec b$, we have the system
            \begin{equation*}
                \systeme{1-\lambda=5\mu,2\lambda=1-\mu}
            \end{equation*}
            which has solution $\lambda = \dfrac49$ and $\mu = \dfrac19$. Thus,
            \begin{align*}
                \overrightarrow{OM} &= \vec b + \dfrac19 (5\vec a - \vec b)\\
                &= \dfrac59 \vec a + \dfrac89 \vec b
            \end{align*}

            \boxt{
                $\overrightarrow{OM} = \dfrac59 \vec a + \dfrac89 \vec b$
            }

        \part
            \begin{center}
                \begin{tikzpicture}
                    \coordinate[label=below:$O$] (O) at (0, 0);
                    \coordinate[label=below:$A$] (A) at (1, 0);
                    \coordinate[label=above left:$B$] (B) at (1, 2);
                    \coordinate[label=below:$C$] (C) at (5, 0);
                    \coordinate[label=above:$D$] (D) at (2, 4);
                    \coordinate[label=above right:$M$] (M) at (13/9, 16/9);
                    \coordinate[label=below:$F$] (F) at (13/9, 0);

                    \draw (O)--(C);
                    \draw (O)--(D);
                    \draw (B)--(C);
                    \draw (A)--(D);
                    \draw[dotted] (F)--(M);
                \end{tikzpicture}
            \end{center}
            
            \noindent Let $F$ be the foot of the perpendicular of $M$ to $OC$. Observe that $\overrightarrow{OM} = \dfrac59 \vec a + \dfrac89 \vec b - \vec a = -\dfrac49 \vec a + \dfrac89 \vec b$ and $\overrightarrow{AC} = 5\vec a - \vec a = 4\vec a$.
            \begin{alignat*}{2}
                &&\abs{\overrightarrow{AM} \times \overrightarrow{AC}} &= 2 \area \triangle AMC\\
                \implies&&\abs{\left(-\dfrac49 \vec a + \dfrac89 \vec b\right) \times 4\vec a} &= 2 \cdot \dfrac12 \cdot FM \cdot AC\\
                \implies&&\abs{-\dfrac{16}9\vec a \times \vec a + 4 \cdot \dfrac89 \vec b \times \vec a} &= FM \cdot \abs{4 \vec a}\\
                \implies&&4 \cdot \dfrac89 \cdot \abs{\vec b \times \vec a} &= 4 \cdot FM\\
                \implies&&FM &= \dfrac89 \abs{\vec b \times \vec a}\\
                && &=  \dfrac89 \abs{\vec a \times \vec b}
            \end{alignat*}

            \boxt{
                The shortest distance of $M$ from the line $OC$ is $\dfrac89 \abs{\vec a \times \vec b}$ units.
            }

        \begin{equation*}
            Y = 3Z
        \end{equation*}

    \problem{RI Prelim 9758/2017/02/Q1}
        Referred to the origin $O$, the points $A$, $B$ and $C$ have position vectors $\vec a$, $\vec b$ and $\vec c$ respectively such that
        \begin{align*}
            \vec a &= 2\vec i + 3\vec j - \vec k\\
            \vec b &= 5\vec i - 2\vec j + 3\vec k\\
            \vec c &= 4\vec i + \vec j - 2\vec k
        \end{align*}

        \begin{enumerate}
            \item Given that $M$ is the mid-point of $AC$, use a vector product to find the exact area of triangle $ABM$.
            \item Find the position vector of the point $N$ on the line $AB$ such that $\overrightarrow{MN}$ is perpendicular to $\overrightarrow{AB}$.
        \end{enumerate}

    \solution
        We have $\vec a = \cveciii23{-1}$, $\vec b = \cveciii5{-2}3$ and $\vec c = \cveciii41{-2}$.

        \part
            By the Midpoint Theorem, $\overrightarrow{OM} = \dfrac12 (\vec a + \vec c) = \cveciii32{-3/2}$. Thus, $\overrightarrow{AM} = \cveciii1{-1}{-1/2}$. We also have $\overrightarrow{AB} = \cveciii3{-5}4$. Hence,
            \begin{align*}
                \area \triangle ABM &= \dfrac12 \abs{\overrightarrow{AM} \times \overrightarrow{AB}}\\
                &= \dfrac12 \abs{\cveciii1{-1}{-1/2} \times \cveciii3{-5}4}\\
                &= \dfrac14 \abs{\cveciii{-2}2{1} \times \cveciii3{-5}4}\\
                &= \dfrac12 \abs{\cveciii{13}{11}4}\\
                &= \dfrac{\sqrt{306}}4\\
                &= \dfrac{3\sqrt{34}}4
            \end{align*}

            \boxt{
                $\area \triangle ABM = \dfrac{3\sqrt{34}}4$ units$^2$
            }

        \part
            Note that the line $AB$ has vector equation
            \begin{equation*}
                l_{AB} : \vec r = \cveciii23{-1} + \lambda\cveciii3{-5}4, \, \lambda \in \mathbb{R}
            \end{equation*}
            Since $\overrightarrow{MN}$ is perpendicular to $\overrightarrow{AB}$, we have
            \begin{alignat*}{2}
                &&\overrightarrow{MN} \cdot \overrightarrow{AB} &= 0\\
                \implies&&\left(\overrightarrow{ON} - \overrightarrow{OM}\right) \cdot \cveciii3{-5}4 &= 0\\
                \implies&&\left[\cveciii23{-1} + \lambda\cveciii3{-5}4 - \cveciii32{-3/2}\right] \cdot \cveciii3{-5}4 &= 0\\
                \implies&&\left[\cveciii{-1}1{1/2} + \lambda\cveciii3{-5}4\right] \cdot \cveciii3{-5}4 &= 0\\
                \implies&&\cveciii{-1}1{1/2} \cdot \cveciii3{-5}4 + \lambda\cveciii3{-5}4 \cdot \cveciii3{-5}4 &= 0\\
                \implies&&-6 + 50\lambda &= 0\\
                \implies&&\lambda &= \dfrac3{25}
            \end{alignat*}

            Hence, $\overrightarrow{ON} = \cveciii23{-1} + \dfrac3{25}\cveciii3{-5}4 = \dfrac1{25} \cveciii{59}{60}{-13}$.

            }
