\section{Assignment B9}

\begin{problem}
    The curve $C$ is defined parametrically by $x = a(2\cos\t + \cos2\t)$, $y = a(2\sin\t + \sin2\t)$ where $0 \leq \t \leq \pi$ and $a$ is a positive constant.

    \begin{enumerate}
        \item Find the coordinates of the points at which $C$ meets the $x$-axis.
        \item Sketch $C$.
        \item Find the exact total length of $C$.
        \item Find the exact area of the curve surface generated when $C$ is rotated through $2\pi$ radians about the $x$-axis.
    \end{enumerate}
\end{problem}
\begin{solution}
    \begin{ppart}
        When $C$ meets the $x$-axis, $y = 0$. \[y = a(2\sin\t + \sin2\t) =  a\bp{2\sin \t + 2\sin \t \cos} = 2a \sin \t \bp{1 + \cos \t} = 0.\] Thus, $\t = 0$ or $\t = \pi$.

        At $\t = 0$, $x = 3a$. At $\t = \pi$, $x = -a$. Hence, $C$ meets the $x$-axis at $(3a, 0)$ and $(-a, 0)$.
    \end{ppart}
    \begin{ppart}
        \begin{center}\tikzsetnextfilename{322}
            \begin{tikzpicture}[trim axis left, trim axis right]
                \begin{axis}[
                    domain = 0:pi,
                    samples = 101,
                    axis y line=middle,
                    axis x line=middle,
                    xtick = {-2, 6},
                    xticklabels = {$-a$, $3a$},
                    ytick = \empty,
                    xlabel = {$x$},
                    ylabel = {$y$},
                    ymax=6,
                    xmin=-3.5,
                    xmax=6.5,
                    legend cell align={left},
                    legend pos=outer north east,
                    after end axis/.code={
                        \path (axis cs:0,0) 
                            node [anchor=north] {$O$};
                        }
                    ]
                    \addplot[plotRed] ({2*(2*cos(\x r) + cos(2*\x r))}, {2*(2*sin(\x r) + sin(2*\x r))});
        
                    \addlegendentry{$C$};
                \end{axis}
            \end{tikzpicture}
        \end{center}
    \end{ppart}
    \begin{ppart}
        Note that \[\der{x}{\t} = -2a(\sin\t+\sin2\t), \quad \der{y}{\t} = 2a(\cos\t + \cos2\t).\] Hence,
        \begin{gather*}
            \bp{\der{x}{\t}}^2 + \bp{\der{y}{\t}}^2 = (2a)^2 \bp{2 + 2\sin \t \sin 2\t + 2\cos \t \cos 2\t} \\
            = (2a)^2 \bp{2 + 2\cos \t} = (2a)^2 \bs{2 + \bp{4\cos^2 \frac\t2 - 2}} = \bp{4a\cos \frac\t2}^2.
        \end{gather*}
        Thus, \[\length = \int_0^{\pi} \sqrt{\bp{\der{x}{\t}}^2 + \bp{\der{y}{\t}}^2} \d \t = 4a \int_0^\pi \cos \frac\t2 \d t = 4a \evalint{2\sin \frac\t2}0\pi = 8a \units.\]
    \end{ppart}
    \begin{ppart}
        \begin{gather*}
            \area = 2\pi \int_0^\pi y \sqrt{\bp{\der{x}{\t}}^2 + \bp{\der{y}{\t}}^2} \d \t = 16\pi a^2 \int_0^\pi \sin \t \bp{1 + \cos \t} \cos \frac\t2 \d \t\\
            = 16\pi a^2 \int_0^\pi \bp{2\sin\frac\t2 \cos \frac\t2}\bp{2\cos^2 \frac\t2} \cos \frac\t2 \d \t = 64 \pi a^2 \int_0^\pi \cos^4 \frac\t2 \sin \frac\t2 \d \t\\
            = -128\pi a^2 \evalint{-\frac{\cos[5]{\t/2}}5}0\pi = \frac{128}5 \pi a^2 \units[2].
        \end{gather*}
    \end{ppart}
\end{solution}

\begin{problem}
    The curve $C$ is given by the equation $y = \frac12(\e^x + \e^{-x})$.

    \begin{enumerate}
        \item Sketch the curve $C$.
        \item Find the exact area bounded by $C$, the lines $x = 2$ and $x = -2$ and the $x$-axis.
        \item Points $A$ and $B$ are on $C$ where $x = 2$ and $x = -2$ respectively. Find the exact length of the arc $AB$.
    \end{enumerate}

    A solid, made of a certain material, is of the shape obtained by rotating the region bounded by $C$, the lines $x = 2$ and $x = -2$ and the $x$-axis about the $y$-axis through $\pi$ radians.

    \begin{enumerate}
        \setcounter{enumi}{3}
        \item Find the exact amount of material required to make this solid if $x$ is measured in cm.
        \item The solid is painted with a brush that uses 2 cm$^3$ of paint for every cm$^2$ of surface painted. Find the exact amount of paint required.
    \end{enumerate}
\end{problem}
\begin{solution}
    \begin{ppart}
        \begin{center}\tikzsetnextfilename{323}
            \begin{tikzpicture}[trim axis left, trim axis right]
                \begin{axis}[
                    domain = -2.5:2.5,
                    samples = 101,
                    axis y line=middle,
                    axis x line=middle,
                    xtick = {2, -2},
                    ytick = \empty,
                    ymin=0,
                    xlabel = {$x$},
                    ylabel = {$y$},
                    legend cell align={left},
                    legend pos=outer north east,
                    after end axis/.code={
                        \path (axis cs:0,0) 
                            node [anchor=north] {$O$};
                        }
                    ]
                    \addplot[plotRed] {0.5 * (e^x + e^(-x))};
        
                    \addlegendentry{$C$};

                    \draw[dotted] (2, 0) -- (2, 3.76);
                    \draw[dotted] (-2, 0) -- (-2, 3.76);

                    \fill (-2, 3.76) circle[radius=2.5pt] node[anchor=west] {$B$};
                    \fill (2, 3.76) circle[radius=2.5pt] node[anchor=east] {$A$};
                \end{axis}
            \end{tikzpicture}
        \end{center}
    \end{ppart}
    \begin{ppart}
        Note that $y = \frac12 (\e^x + \e^{-x}) = \cosh x$ is an even function. Hence,
        \begin{gather*}
            \area = \int_{-2}^2 y \d x = 2 \int_0^2 \cosh x \d x = 2 \evalint{\sinh x}02 \\
            = 2 \bp{\sinh 2 - \sinh 0} = 2 \bp{\frac{e^2 - e^{-2}}{2} - 0} = \e^2 - \e^{-2} \units[2].
        \end{gather*}
    \end{ppart}
    \begin{ppart}
        Note that \[\der{y}{x} = \der{}{x} \cosh x = \sinh x,\] whence \[\sqrt{1 + \bp{\der{y}{x}}^2} = \sqrt{1 + \sinh^2 x} = \sqrt{\cosh^2 x} = \cosh x\] Hence, \[\length = \int_{-2}^2 \sqrt{1 + \bp{\der{y}{x}}^2} \d x = \int_{-2}^2 \cosh x \d x = 2 \int_0^2 \cosh x \d x = \e^2 - \e^{-2} \units.\]
    \end{ppart}
    \begin{ppart}
        We have \[\volume = 2\pi \int_0^2 xy \d x = 2\pi \int_0^2 x \cosh x \d x.\] Integrating by parts, 
        \[\begin{array}{r c @{\hspace*{1.0cm}} c}\toprule
            & D & I \\\cmidrule{1-3}
            + & x & \cosh x\\
            - & 1 & \sinh x \\
            + & 0 & \cosh x\\\bottomrule
        \end{array}\] Thus,
        \begin{gather*}
            \volume = 2\pi \evalint{x\sinh x - \cosh x}02 = 2\pi \bs{\bp{2\sinh 2 - \cosh 2} - \bp{0\sinh 0 - \cosh 0}} \\
            = 2\pi \bs{2 \bp{\frac{e^2 - e^2}2} - \frac{e^2 + e^{-2}}2 + 1} = \pi \bp{e^2 - 3e^{-2} + 2}.
        \end{gather*}

        Thus, $\pi \bp{e^2 - 3e^{-2} + 2}$ cm$^3$ of material is required.
    \end{ppart}
    \begin{ppart}
        \begin{gather*}
            \area = \text{Area of curved surface} + \text{Area of side} + \text{Area of bottom}\\
            = 2\pi \int_0^2 x\cosh x \d x + 2^2\pi + 2^2\pi\cosh 2 = \pi \bp{e^2 - 3e^{-2} + 2} + 4\pi + 4\pi \bp{\frac{e^2 + e^{-2}}2}\\
            = \pi \bs{3e^2 - e^{-2} + 6}.
        \end{gather*}

        Thus, $2\pi \bp{3e^2 - e^{-2} + 6}$ cm$^3$ of paint is required.
    \end{ppart}
\end{solution}
