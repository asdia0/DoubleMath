    \problem{}
        Find

        \begin{enumerate}
            \item $\displaystyle\int \dfrac1{\sqrt{3-2x}}  \d x$
            \item $\displaystyle\int \dfrac1{3-2x}  \d x$
            \item $\displaystyle\int \dfrac1{3-2x^2}  \d x$
            \item $\displaystyle\int \dfrac1{\sqrt{3-2x^2}}  \d x$
            \item $\displaystyle\int \dfrac{x}{\sqrt{3-2x^2}}  \d x$
            \item $\displaystyle\int \dfrac1{3+4x+2x^2}  \d x$
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int \dfrac1{\sqrt{3-2x}}  \d x &= -\dfrac12 \int \dfrac1{\sqrt u}  \d u \usub{u &= 3 - 2x \\ \d u &= -2x  \d x\\ \d x &= -1/2  \d u}\\
                &= -\dfrac12 \cdot 2 u^{1/2} + C\\
                &= -u^{1/2} + C\\
                &= -\sqrt{3-2x} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{\sqrt{3-2x}}  \d x = -\sqrt{3-2x} + C$}

        \part
            \begin{align*}
                \int \dfrac1{3-2x}  \d x &= -\dfrac12 \int \dfrac1{u}  \d u \usub{u &= 3 - 2x \\ \d u &= -2x  \d x\\ \d x &= -1/2  \d u}\\
                &= -\dfrac12 \ln \abs{u} + C\\
                &= -\dfrac12 \ln \abs{3 - 2x} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{3-2x}  \d x = -\dfrac12 \ln \abs{3 - 2x} + C$}

        \part
            \begin{align*}
                \int \dfrac1{3-2x^2}  \d x &= \dfrac12 \int \dfrac1{3/2 - x^2}  \d x\\
                &= \dfrac12 \cdot \bp{1 \Big/ 2 \sqrt{3/2}} \ln{\dfrac{\sqrt{3/2} + x}{\sqrt{3/2} - x}} + C\\
                &= \dfrac1{2\sqrt6} \ln{\dfrac{\sqrt3 + \sqrt2 x}{\sqrt3 - \sqrt2 x}} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{3-2x^2}  \d x = \dfrac1{2\sqrt6} \ln{\dfrac{\sqrt3 + \sqrt2 x}{\sqrt3 - \sqrt2 x}} + C$}

        \part
            \begin{align*}
                \int \dfrac1{\sqrt{3-2x^2}}  \d x &= \dfrac1{\sqrt2} \int \dfrac1{\sqrt{3/2 - x^2}}  \d x\\
                &= \dfrac1{\sqrt2} \arcsin{x \Big/ \sqrt{3/2}} + C\\
                &= \dfrac{\sqrt2}2 \arcsin{\dfrac{\sqrt6 x}{3}} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{\sqrt{3-2x^2}}  \d x = \dfrac{\sqrt2}2 \arcsin{\dfrac{\sqrt6 x}{3}} + C$}

        \part
            \begin{align*}
                \int \dfrac{x}{\sqrt{3-2x^2}}  \d x &= -\dfrac14 \int \dfrac1{\sqrt {u}} \d u \usub{u &= 3-2x^2 \\ \d u &= -4x  \d x \\ x\d x &= -1/4  \d u}\\
                &= -\dfrac14 \cdot 2 u^{1/2} + C\\
                &= -\dfrac12 \sqrt{3 - 2x^2} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac{x}{\sqrt{3-2x^2}}  \d x = -\dfrac12 \sqrt{3 - 2x^2} + C$}

        \part
            \begin{align*}
                \int \dfrac1{3+4x+2x^2}  \d x &= \int \dfrac1{2(x+1)^2 + 1}  \d x\\
                &= \dfrac12 \int \dfrac1{(x+1)^2 + 1/2}  \d x\\
                &= \dfrac12 \cdot \bp{1 \Big/ {\sqrt{1/2}}} \arctan{\dfrac{x+1}{1 \Big/ \sqrt{1/2}}} + C\\
                &= \dfrac{\sqrt2}2 \arctan{\sqrt2 (x + 1)} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{3+4x+2x^2}  \d x = \dfrac{\sqrt2}2 \arctan{\sqrt2 (x + 1)} + C$}

    \problem{}
        Find

        \begin{enumerate}
            \item $\displaystyle\int \dfrac{\sec^2 3x}{\tan 3x}  \d x$
            \item $\displaystyle\int \cos (3x + \a)  \d x$, where $\a$ is a constant
            \item $\displaystyle\int \cos^2 3x  \d x$
            \item $\displaystyle\int e^{1-2x}  \d x$
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int \dfrac{\sec^2 3x}{\tan 3x}  \d x &= \int \dfrac{1 + \tan^2 3x}{\tan 3x}  \d x\\
                &= \int \bp{\cot 3x + \tan 3x} \d x\\
                &= \dfrac13 \ln \sin 3x + \dfrac13 \ln \sec 3x + C\\
                &= \dfrac13 \bp{\ln \sin 3x + \ln \sec 3x } + C\\
                &= \dfrac13 \ln{\sin 3x \sec 3x} + C\\
                &= \dfrac13 \ln \tan 3x + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac{\sec^2 3x}{\tan 3x}  \d x = \dfrac13 \ln \tan 3x + C$}

        \part
            \begin{align*}
                \int \cos (3x + \a)  \d x &= \dfrac13 \int \cos u  \d u \usub{u &= 3x + \a \\ \d u &= 3 \d x\\\d x &= 1/3  \d u}\\
                &= \dfrac13 \sin u + C\\
                &= \dfrac13 \sin(3x + \a) + C
            \end{align*}

            \boxt{$\displaystyle\int \cos (3x + \a)  \d x = \dfrac13 \sin(3x + \a) + C$}

        \part
            Recall that $cos^2\t = \dfrac{1 + \cos2\t}2$. Hence, $\cos^2 3x = \dfrac{1 + \cos 6x}2$.

            \begin{align*}
                \int \cos^2 3x  \d x &= \int \dfrac{1 + \cos 6x}2  \d x\\
                &= \dfrac12 \int (1 + \cos6x)  \d x\\
                &= \dfrac12 \bp{x + \dfrac16 \sin6x} + C\\
                &= \dfrac12 x + \dfrac1{12} \sin6x + C
            \end{align*}

            \boxt{$\displaystyle\int \cos^2 3x  \d x = \dfrac12 x + \dfrac1{12} \sin6x + C$}

        \part
            \begin{align*}
                \int e^{1-2x}  \d x &= -\dfrac12 \int e^u \d u \usub{u &= 1-2x \\ \d u &= -2\d x \\ \d x &= -1/2  \d u}\\
                &= -\dfrac12 e^u + C\\
                &= -\dfrac12 e^{1-2x} + C
            \end{align*}

            \boxt{$\displaystyle\int e^{1-2x}  \d x = -\dfrac12 e^{1-2x} + C$}

    \problem{}
        Find

        \begin{enumerate}
            \item $\displaystyle\int 2x\sqrt{3x^2 - 5}  \d x$
            \item $\displaystyle\int \dfrac{x^2 - 1}{\sqrt{x^3 - 3x}}  \d x$
            \item $\displaystyle\int \sin x \sqrt{\cos x}  \d x$
            \item $\displaystyle\int e^{2x} (1 - e^{2x})^4  \d x$
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int 2x\sqrt{3x^2 - 5}  \d x &= \dfrac16 \int 2\sqrt{u}  \d u \usub{u &= 3x^2 - 5 \\ \d u &= 6x  \d x \\ x \d x &= 1/6  \d u}\\
                &= \dfrac13 \int \sqrt{u}  \d u\\
                &= \dfrac13 \cdot \dfrac23 u^{3/2} + C\\
                &= \dfrac29 u^{3/2} + C\\
                &= \dfrac29 \bp{3x^2 - 5}^{3/2} + C
            \end{align*}

            \boxt{$\displaystyle\int 2x\sqrt{3x^2 - 5}  \d x = \dfrac29 \bp{3x^2 - 5}^{3/2} + C$}
            
        \part
            \begin{align*}
                \int \dfrac{x^2 - 1}{\sqrt{x^3 - 3x}}  \d x &= \dfrac13 \int \dfrac{\d u}{\sqrt{u}} \usub{u &= x^3 - 3x\\\d u &= 3x^2 - 3 \d x\\x^2 - 1  \d x &= 1/3 \d u}\\
                &= \dfrac13 \bp{ 2u^{1/2} } + C\\
                &= \dfrac23 \sqrt{x^3 - 3x} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac{x^2 - 1}{\sqrt{x^3 - 3x}}  \d x = \dfrac23 \sqrt{x^3 - 3x} + C$}

        \part
            \begin{align*}
                \int \sin x \sqrt{\cos x}  \d x &= - \int \sqrt{u}  \d u \usub{u &= \cos x\\\d u &= -\sin x  \d x\\\sin x  \d x &= -\d u}\\
                &= -\dfrac23 u^{3/2} + C\\
                &= -\dfrac23 (\cos x)^{3/2} + C
            \end{align*}

            \boxt{$\displaystyle\int \sin x \sqrt{\cos x}  \d x = -\dfrac23 (\cos x)^{3/2} + C$}

        \part
            \begin{align*}
                \int e^{2x}(1 - e^{2x})^4  \d x &= -\dfrac12 \int u^4  \d u \usub{u &= 1 - e^{2x}\\\d u &= -2e^{2x}  \d x\\e^{2x}  \d x &= -1/2 \d u}\\
                &= -\dfrac12 \cdot \dfrac{u^5}{5} + C\\
                &= -\dfrac1{10} (1 - e^{2x})^5 + C
            \end{align*}

            \boxt{$\displaystyle\int e^{2x}(1 - e^{2x})^4  \d x = -\dfrac1{10} (1 - e^{2x})^5 + C$}

    \problem{}
        Find

        \begin{enumerate}
            \item $\displaystyle\int \dfrac1{\sqrt{x}(1- \sqrt{x})}  \d x$
            \item $\displaystyle\int \dfrac{3x}{x+3}  \d x$
            \item $\displaystyle\int \dfrac{\sin x + \cos x}{\sin x - \cos x}  \d x$
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int \dfrac1{\sqrt{x}(1- \sqrt{x})}  \d x &= -2 \int \dfrac1{u}  \d u\usub{u &= 1 - \sqrt{x}\\\d u &= -1/(2\sqrt x)  \d x\\ 1/\sqrt{x}  \d x &= -2  \d u}\\
                &= -2 \ln \abs{u} + C\\
                &= -2 \ln \abs{1 - \sqrt{x}} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{\sqrt{x}(1- \sqrt{x})}  \d x = -2 \ln \abs{1 - \sqrt{x}} + C$}

        \part
            \begin{align*}
                \int \dfrac{3x}{x+3}  \d x &= \int \bp{3 - \dfrac9{x+3} }\d x\\
                &= 3x - 9\ln\abs{x+3} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac{3x}{x+3}  \d x = 3x - 9\ln\abs{x+3} + C$}

        \part
            \begin{align*}
                \int \dfrac{\sin x + \cos x}{\sin x - \cos x}\d x &= \int \dfrac1{u}  \d u \usub{u &= \sin x - \cos x\\\d u &= \cos x + \sin x  \d x}\\
                &= \ln \abs{u} + C\\
                &= \ln \abs{\sin x - \cos x} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac{\sin x + \cos x}{\sin x - \cos x}\d x = \ln \abs{\sin x - \cos x} + C$}

    \problem{}
        Find 

        \begin{enumerate}
            \item $\displaystyle\int \dfrac{e^{-\sqrt{x}}}{\sqrt{x}}  \d x$
            \item $\displaystyle\int (\sin x)(\cos x)(e^{\cos 2x})  \d x$
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int \dfrac{e^{-\sqrt{x}}}{\sqrt{x}}  \d x &= -2\int e^{u}  \d u \usub{u &= -\sqrt{x}\\\d u &= -1/(2\sqrt{x})  \d x\\1/\sqrt{x}  \d x &= -2  \d u}\\
                &= -2e^u + C\\
                &= -2e^{-\sqrt{x}} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac{e^{-\sqrt{x}}}{\sqrt{x}}  \d x = -2e^{-\sqrt{x}} + C$}

        \part
            \begin{align*}
                \int (\sin x)(\cos x)(e^{\cos 2x})  \d x &= \dfrac12 \int \sin 2x \cdot e^{\cos 2x} \d x\\ &= -\dfrac14 \int e^u  \d u \usub{u &= \cos 2x\\\d u &= -2\sin 2x  \d x\\\sin 2x  \d x &= -1/2  \d u}\\
                &= -\dfrac14 e^u + C\\
                &= -\dfrac14 e^{\cos 2x} + C
            \end{align*}

            \boxt{$\displaystyle\int (\sin x)(\cos x)(e^{\cos 2x})  \d x = -\dfrac14 e^{\cos 2x} + C$}

    \problem{}
        Find

        \begin{enumerate}
            \item $\displaystyle\int \tan^2 2x  \d x$
            \item $\displaystyle\int \dfrac1{1 + \cos 2t}  \d t$
            \item $\displaystyle\int \sin{\dfrac52\t}\cos{\dfrac12\t}  d\t$
            \item $\displaystyle\int \tan^4 x  \d x$
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int \tan^2 2x  \d x &= \int \bp{\sec^2 2x - 1}  \d x\\
                &= \dfrac12 \tan 2x - x + C
            \end{align*}

            \boxt{$\displaystyle\int \tan^2 2x  \d x = \dfrac12 \tan 2x - x + C$}

        \part
            \begin{align*}
                \int \dfrac1{1 + \cos 2t}  \d t &= \int \dfrac1{1 + (2\cos^2 t - 1)}  \d t\\
                &= \dfrac12 \int \dfrac1{\cos^2 t}  \d t\\
                &= \dfrac12 \int \sec^2 t  \d t\\
                &= \dfrac12 \tan t + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{1 + \cos 2t}  \d t = \dfrac12 \tan t + C$}

        \part
            \begin{align*}
                \int \sin{\dfrac52\t}\cos{\dfrac12\t}  d\t &= \dfrac12 \int \bp{\sin{\dfrac52 \t + \dfrac12 \t} + \sin{\dfrac52\t - \dfrac12 \t}}  d\t\\
                &= \dfrac12 \int \bp{\sin 3\t + \sin2\t}  d\t\\
                &= \dfrac12 \bp{ -\dfrac13 \cos3\t - \dfrac12 \cos2\t } + C\\
                &= -\dfrac16 \cos 3\t - \dfrac14 \cos 2\t + C
            \end{align*}

            \boxt{$\displaystyle\int \sin{\dfrac52\t}\cos{\dfrac12\t}  d\t = -\dfrac16 \cos 3\t - \dfrac14 \cos2\t + C$}

        \part
            Note that
            \[
                \int \tan^2 x  \d x = \tan x - x + C
            \]

             and 
            \begin{align*}
                \int \tan^2 x \sec^2 x  \d x &= \int u^2  \d u \usub{u &= \tan x\\\d u &= \sec^2 x  \d x}\\
                &= \dfrac13 u^3 + C\\
                &= \dfrac13 \tan^3 x + C
            \end{align*}

            Hence,
            \begin{align*}
                \int \tan^4 x  \d x &= \int \tan^2 x \cdot \tan^2 x  \d x\\
                &= \int \tan^2 x \bp{\sec^2 x - 1}  \d x\\
                &= \int \bp{\tan^2 x \sec^2 x - \tan^2 } \d x\\
                &= \dfrac13 \tan^3 x - \bp{\tan x - x} + C\\
                &= \dfrac13 \tan^3 x - \tan x + x + C
            \end{align*}

            \boxt{$\displaystyle\int \tan^4 x  \d x = \dfrac13 \tan^3 x - \tan x + x + C$}

    \problem{}
        Find

        \begin{enumerate}
            \item $\displaystyle\int \dfrac1{4x^2 + 2x + 10}  \d x$
            \item $\displaystyle\int \dfrac{x^2}{1 - x^2}  \d x$
            \item $\displaystyle\int \dfrac1{\sqrt{3 + 2x - x^2}}  \d x$
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int \dfrac1{4x^2 + 2x + 10}  \d x &= \int \dfrac1{\bp{2x + 1/2}^2 + 39/4}  \d x\\
                &= 4 \int \dfrac1{\bp{4x + 1}^2 + 39}  \d x\\
                &= 4 \cdot \dfrac14 \cdot \dfrac1{\sqrt{39}} \arctan \bp{\dfrac{4x+1}{\sqrt{39}}} + C\\
                &= \dfrac1{\sqrt{39}} \arctan \bp{\dfrac{4x+1}{\sqrt{39}}} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{4x^2 + 2x + 10}  \d x = \dfrac1{\sqrt{39}} \arctan \bp{\dfrac{4x+1}{\sqrt{39}}} + C$}

        \part
            \begin{align*}
                \int \dfrac{x^2}{1 - x^2}  \d x &= \int \bp{\dfrac1{1 - x^2} - 1} \d x\\
                &= \dfrac1{2}\ln\bp{\dfrac{1 + x}{1 - x}} - x + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac{x^2}{1 - x^2}  \d x = \dfrac1{2}\ln\bp{\dfrac{1 + x}{1 - x}} - x + C$}

        \part
            \begin{align*}
                \int \dfrac1{\sqrt{3 + 2x - x^2}}  \d x &= \int \dfrac1{\sqrt{2^2 - (x-1)^2}}  \d x\\
                &= \arcsin{\dfrac{x-1}{2}} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{\sqrt{3 + 2x - x^2}}  \d x = \arcsin{\dfrac{x-1}{2}} + C$}

    \problem{}
        Evaluate the following without the use of graphic calculator:

        \begin{enumerate}
            \item $\displaystyle\int_{\frac13 \pi}^{\frac23 \pi} 4\cot\dfrac{x}2 \csc^2\dfrac{x}2  \d x$
            \item $\displaystyle\int_0^4 \dfrac{x+2}{\sqrt{2x+1}}  \d x$
            \item $\displaystyle\int_0^1 \dfrac2{(1+x)(1+x^2)} \d x$
            \item $\displaystyle\int_{-4}^{-2} \dfrac{x^3 + 2}{x^2 - 1}  \d x$
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int_{\frac13 \pi}^{\frac23 \pi} 4\cot\dfrac{x}2 \csc^2\dfrac{x}2  \d x &= -4\int_{\frac13 \pi}^{\frac23 \pi} -\cot\dfrac{x}2 \csc\dfrac{x}2 \cdot \csc{\dfrac12 x}  \d x\usub{u &= \csc \tfrac{x}2\\\d u &= -\tfrac12\cot\tfrac{x}2 \csc\tfrac{x}2\\2\d u &= -\cot\tfrac{x}2 \csc\tfrac{x}2\\x = \tfrac13\pi &\implies u = 2\\x = \tfrac23\pi &\implies u = \tfrac2{\sqrt3}}\\
                &= -8 \int_{2}^{\tfrac2{\sqrt3}} u  \d u\\
                &= -8 \evalint{\dfrac12 u^2}{2}{\tfrac2{\sqrt3}}\\
                &= \dfrac{32}3
            \end{align*}

            \boxt{$\displaystyle\int_{\frac13 \pi}^{\frac23 \pi} 4\cot\dfrac{x}2 \csc^2\dfrac{x}2  \d x = \dfrac{32}3$}

        \part
            \begin{align*}
                \int_0^4 \dfrac{x+2}{\sqrt{2x+1}}  \d x &= \int_0^4 \dfrac{\tfrac12 (2x+1) + \tfrac32}{\sqrt{2x+1}}  \d x\\
                &= \dfrac12 \int_0^4 \bp{\dfrac{2x+1}{\sqrt{2x+1}} + \dfrac3{\sqrt{2x +1}}}  \d x\\
                &= \dfrac12 \int_0^4 \bp{\sqrt{2x+1} + \dfrac3{\sqrt{2x +1}}}  \d x \usub{u &= 2x+1\\\d u &= 2\d x\\\d x &= 1/2 \d u\\x = 0 &\implies u = 1\\x = 4 &\implies u = 9}\\
                &= \dfrac14 \int_1^9 \bp{\sqrt{u} + \dfrac3{\sqrt{u}}}  \d u\\
                &= \dfrac14 \evalint{\dfrac{u^{3/2}}{3/2} + \dfrac{3u^{1/2}}{1/2}}{1}{9}\\
                &= \dfrac{22}3
            \end{align*}

            \boxt{$\displaystyle\int_0^4 \dfrac{x+2}{\sqrt{2x+1}}  \d x = \dfrac{22}3$}

        \part
            \begin{align*}
                \int_0^1 \dfrac2{(1+x)(1+x^2)} \d x &= \int_0^1 \bp{\dfrac1{1+x} + \dfrac1{1+x^2} - \dfrac{x}{1 + x^2}}\d x\\
                &= \evalint{\ln \abs{1 + x}}01 + \evalint{\arctan x}01 - \int_0^1 \dfrac{x}{1 + x^2}  \d x\usub{u &= 1+x^2\\\d u&=2x\d x\\x\d x&= 1/2 \d u\\x = 0 &\implies u = 1\\x = 1 &\implies u = 2}\\
                &= \ln 2 + \dfrac{\pi}4 - \dfrac12 \int_1^2 \dfrac1{u} \d u\\
                &= \ln 2 + \dfrac{\pi}4 - \dfrac12 \evalint{\ln \abs{u}}{1}{2}\\
                &= \ln 2 + \dfrac{\pi}4 - \dfrac12 \ln 2\\
                &= \dfrac12 \ln 2 + \dfrac\pi4
            \end{align*}

            \boxt{$\displaystyle\int_0^1 \dfrac2{(1+x)(1+x^2)} \d x = \dfrac12\ln2 + \dfrac\pi4$}

        \part
            \begin{align*}
                \int_{-4}^{-2} \dfrac{x^3 + 2}{x^2 - 1}  \d x &= \int_{-4}^{-2} \bp{x + \dfrac{x+2}{x^2 - 1}} \d x\\
                &= \evalint{\dfrac{x^2}2}{-4}{-2} + \int_{-4}^{-2} \dfrac{x+2}{(x-1)(x+1)} \d x\\
                &= -6 + \int_{-4}^{-2} \bp{\dfrac{3/2}{x-1} - \dfrac{1/2}{x+1}} \d x\\
                &= -6 + \evalint{\dfrac32 \ln \abs{x-1} - \dfrac12 \ln \abs{x+1}}{-4}{-2}\\
                &= -6 + 2\ln3 - \dfrac32 \ln 5
            \end{align*}

            \boxt{$\displaystyle\int_{-4}^{-2} \dfrac{x^3 + 2}{x^2 - 1}  \d x = -6 + 2\ln3 - \dfrac32 \ln 5$}

    \problem{}
        Using the given substitution, find

        \begin{enumerate}
            \item $\displaystyle\int \dfrac{x}{(2x+3)^3}  \d x$ \hfill $[u = 2x+3]$
            \item $\displaystyle\int \dfrac1{e^x + 4e^{-x}}  \d x$ \hfill $[u = e^x]$
            \item $\displaystyle\int_0^{\sqrt2} \sqrt{4 - y^2}  \d y$ \hfill $[y = 2\sin\t]$
            \item $\displaystyle\int_0^{\pi/2} \dfrac1{1 + \sin\t}  \d \t$ \hfill $\left[t = \tan \dfrac\t2\right]$
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int \dfrac{x}{(2x+3)^3}  \d x &= \dfrac14 \int \dfrac{u-3}{u^3}  \d x \usub{u &= 2x + 3\\x &= (u-3)/2\\d x &= 1/2 \d u}\\
                &= \dfrac14 \int \bp{\dfrac1{u^2} - \dfrac3{u^3}} \d u\\
                &= \dfrac14 \bp{\dfrac{u^{-1}}{-1} - \dfrac{3u^{-2}}{-2} } + C\\
                &= \dfrac38 (2x+3)^{-2} - \dfrac14 (2x+3)^{-1} + C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac{x}{(2x+3)^3}  \d x = \dfrac38 (2x+3)^{-2} - \dfrac14 (2x+3)^{-1} + C$}

        \part
            \begin{align*}
                \int \dfrac1{e^x + 4e^{-x}}  \d x &= \int \dfrac{e^x}{e^{2x} + 4} \d x \\
                &= \int \dfrac{u}{u^2 + 4} \cdot \dfrac1{u} \d u \usub{u &= e^x\\\ln u &= x\\d x &= 1/u \d u}\\
                &= \int \dfrac1{u^2 + 2^2} \d u\\
                &= \dfrac12 \arctan \bp{\dfrac{u}2} + C\\
                &= \dfrac12 \arctan \bp{\dfrac{e^x}2}+ C
            \end{align*}

            \boxt{$\displaystyle\int \dfrac1{e^x + 4e^{-x}}  \d x = \dfrac12 \arctan \bp{\dfrac{e^x}2}+ C$}

        \part
            \begin{align*}
                \int_0^{\sqrt2} \sqrt{4 - y^2}  \d y &= 2\int_0^{\pi/4} \cos\t\sqrt{4 - (2\sin\t)^2} \d \t \usub{y &= 2\sin\t\\\d y &= 2 \cos\t \d \t\\y = 0 &\implies \t=0\\y = \sqrt{2} &\implies \t = \pi /4}\\
                &= 2\int_0^{\pi/4} \cos\t\sqrt{4 - 4\sin^2\t}\d \t\\
                &= 4\int_0^{\pi/4} \cos\t\sqrt{1 - \sin^2\t}\d \t\\
                &= 4\int_0^{\pi/4} \cos^2\t \d \t\\
                &= 4\int_0^{\pi/4} \dfrac{1 + \cos 2\t}{2} \d \t\\
                &= 2 \evalint{\t + \dfrac12 \sin2\t}{0}{\pi/4}\\
                &= 1 + \dfrac\pi2
            \end{align*}

                &= 1
            \end{align*}

            \boxt{$\displaystyle\int_0^{\pi/2} \dfrac1{1 + \sin\t}  \d \t = 1$}

        \begin{enumerate}
            \item Solve the inequality $x^2 + 2x - 3 < 0$.
            \item Without using the graphing calculator, evaluate
            \begin{enumerate}
                \item $\displaystyle\int_{-4}^4 \abs{x^2 + 2x - 3} \d x$
                \item $\displaystyle\int_0^2 x\abs{x^2 + 2x - 3} \d x$
            \end{enumerate}
        \end{enumerate}

        \part
            \begin{alignat*}{2}
                &&x^2 + 2x - 3 &< 0\\
                \implies&&(x+1)^2 - 4 &< 0\\
                \implies&&(x+1)^2 &< 4\\
                \implies&&-2 < x+1 < 2\\
                \implies&&-3 < x < 1
            \end{alignat*}

            \boxt{$-3 < x < 1$}

        \part
            \subpart
                Let $F(x) = \int \bp{x^2 + 2x - 3}\d x = \dfrac13 x^3 + x^2 - 3x + C$. Then,
                \begin{align*}
                    &\int_{-4}^4 \abs{x^2 + 2x - 3} \d x\\
                    &\mspace{25 mu}= \int_{-4}^{-3} \abs{x^2 + 2x - 3} \d x + \int_{-3}^1 \abs{x^2 + 2x - 3} \d x + \int_1^4 \abs{x^2 + 2x - 3} \d x\\

                    &\mspace{25 mu}= \Big(F(-3) - F(-4)\Big) - \Big(F(1) - F(-3)\Big) + \Big(F(4) - F(1)\Big)\\
                    &\mspace{25 mu}= 40
                \end{align*}

                \boxt{$\displaystyle\int_{-4}^4 \abs{x^2 + 2x - 3} \d x = 40$}

            \subpart

                    \int_0^2 x\abs{x^2 + 2x - 3} \d x &= \int_0^1 x\abs{x^2 + 2x - 3} \d x + \int_1^2 x\abs{x^2 + 2x - 3} \d x\\

    \problem{}
        The indefinite integral $\displaystyle\int \dfrac{P(x)}{x^3 + 1}  \d x$, where $P(x)$ is a polynomial in $x$, is denoted by $I$.
        
        \begin{enumerate}
            \item Find $I$ when $P(x) = x^2$.
            \item By writing $x^3 + 1 = (x+1)\bp{x^2 + Ax + B}$, where $A$ and $B$ are constants, find $I$ when
            \begin{enumerate}
                \item $P(x) = x^2 - x + 1$
                \item $P(x) = x + 1$
            \end{enumerate}
            \item Using the results of parts (a) and (b), or otherwise, find $I$ when $P(x) = 1$.
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                \int \dfrac{x^2}{x^3 + 1} \d x &= \dfrac13 \int \dfrac1{u} \d u \usub{u &= x^3 + 1\\\d u &= 3x^2 \d x\\x^2 \d x &= 1/3 \d u}\\
                &= \dfrac13 \ln \abs{u} + C\\
                &= \dfrac13 \ln \abs{x^3 + 1} + C
            \end{align*}

            \boxt{$I = \dfrac13 \ln \abs{x^3 + 1} + C$}

        \part
            \[
                x^3 + 1 = (x+1)\bp{x^2 - x + 1}
            \]

            \subpart
                \begin{align*}
                    \int \dfrac{x^2 - x + 1}{x^3 + 1} \d x &= \int \dfrac{x^2 - x + 1}{(x+1)\bp{x^2 - x + 1}} \d x\\
                    &= \int \dfrac1{x+1} \d x\\
                    &= \ln \abs{x+1} + C
                \end{align*}

                \boxt{$I = \ln\abs{x+1} + C$}

            \subpart
                \begin{align*}
                    \int \dfrac{x+1}{x^3 + 1} \d x &= \int \dfrac{x+1}{(x+1)\bp{x^2 - x + 1}} \d x\\
                    &= \int \dfrac1{x^2 - x + 1} \d x\\
                    &= \int \dfrac1{(x-1/2)^2 + 3/4} \d x\\
                    &= \dfrac1{\sqrt{3/4}} \arctan \bp{ \dfrac{x - 1/2}{\sqrt{3/4}}} + C\\
                    &= \dfrac2{\sqrt3}\arctan \bp{\dfrac{2x-1}{\sqrt3}} + C
                \end{align*}

                \boxt{$I = \dfrac2{\sqrt3}\arctan \bp{\dfrac{2x-1}{\sqrt3}} + C$}

        \part
            Observe that $1 = \dfrac12 \bp{(x^2 - x + 1) - x^2 + (x+1)}$. Hence,

            \begin{align*}
                \int \dfrac1{x^3 + 1} \d x &= \dfrac12\bp{\int\dfrac{x^2 - x + 1}{x^3 + 1} \d x - \int\dfrac{x^2}{x^3 + 1} \d x + \int\dfrac{x+1}{x^3 + 1} \d x}\\
                &= \dfrac12 \bp{\ln \abs{x+1} - \dfrac13 \ln\abs{x^3 + 1} + \dfrac2{\sqrt3}\arctan \bp{\dfrac{2x-1}{\sqrt3}}} + C\\
                &= \dfrac12\ln \abs{x+1} - \dfrac16 \ln\abs{x^3 + 1} + \dfrac1{\sqrt3}\arctan \bp{\dfrac{2x-1}{\sqrt3}} + C
            \end{align*}

            \boxt{$I = \dfrac12\ln \abs{x+1} - \dfrac16 \ln\abs{x^3 + 1} + \dfrac1{\sqrt3}\arctan \bp{\dfrac{2x-1}{\sqrt3}} + C$}
