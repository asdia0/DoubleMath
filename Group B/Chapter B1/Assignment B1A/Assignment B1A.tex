    \problem{}
        The curve $C$ has equation $y = \dfrac{ax^2+bx-2}{x+4}$, where $a$ and $b$ are constants. It is given that $y = 2x-5$ is an asymptote of $C$.

        \begin{enumerate}
            \item Find the values of $a$ and $b$.
            \item Sketch $C$.
            \item Using an algebraic method, find the set of values that $y$ cannot take.
            \item By drawing a sketch of another suitable curve in the same diagram as your sketch of $C$ in part (b), deduce the number od distinct real roots of the equation $x^3 + 6x^2 + 3x - 2 = 0$.
        \end{enumerate}

    \solution
        \part
            Since $y = 2x-5$ is an asymptote of $C$, $\dfrac{ax^2 + bx-2}{x+4}$ can be expressed in the form $2x-5 + \dfrac{k}{x+4}$, where $k$ is a constant.

            \begin{alignat*}{2}
                &&\dfrac{ax^2 + bx-2}{x+4} &= 2x-5 + \dfrac{k}{x+4}\\
                \implies&&ax^2+bx-2 &= (2x-5)(x+4) + k\\
                \implies&&ax^2+bx-2 &= 2x^2+3x-20+k   
            \end{alignat*}

            Comparing coefficients of $x^2$, $x$ and constant terms, we have $a = 2$, $b = 3$ and $k = 18$.

            \boxt{
                $a = 2$, $b = 3$
            }

        \part
            \begin{center}
                \begin{tikzpicture}[trim axis left, trim axis right]
                    \begin{axis}[
                        domain = -10:5,
                        samples = 301,
                        axis y line=middle,
                        axis x line=middle,
                        xtick = {0.5},
                        xticklabels = {$\tfrac12$},
                        ytick = \empty,
                        xlabel = {$x$},
                        ylabel = {$y$},
                        ymin=-40,
                        ymax = 10,
                        legend cell align={left},
                        legend pos=outer north east,
                        after end axis/.code={
                            \path (axis cs:0,0) 
                                node [anchor=south east] {$O$};
                            }
                        ]
                        \addplot[plotRed, unbounded coords=jump] {(2*x^2 + 3*x-2)/(x+4)};
            
                        \addlegendentry{$y = \tfrac{2x^2 + 3x-2}{x+4}$};

                        \addplot[plotBlue] {-x^2};

                        \addlegendentry{$y=-x^2$};

                        \draw[dotted, thick] (-4, 10) -- (-4, -40) node[anchor=south west] {$x = -4$};

                        \fill (-1, -1) circle[radius=2.5 pt];
                        
                        \node[anchor=north] at (-2, -1) {$(-1, -1)$};

                        \fill (-7, -25) circle[radius=2.5 pt];
                        
                        \node[anchor=north] at (-7.5, -26) {$(-7, -25)$};
                        
                        \addplot[dotted, thick] {2*x-5};

                        \node[anchor=south, rotate=22.5] at (-7, -19) {$y= 2x-5$};
                    \end{axis}
                \end{tikzpicture}
            \end{center}
            
        \part
            \begin{alignat*}{2}
                &&y &= \dfrac{2x^2 + 3x - 2}{x+4} \\
                \implies&&(x+4)y &= 2x^2 + 3x -2\\
                \implies&&2x^2 + (3-y)x - (2+4y) &= 0
            \end{alignat*}

            For values that $y$ cannot take on, there exist no solutions to $2x^2 + (3-y)x - (2+4y) = 0$. Hence, $\Delta < 0$.

            \begin{alignat*}{2}
                &&\Delta &< 0\\
                \implies&&(3-y)^2 - 4(2)\left(-(2+4y)\right) &< 0\\
                \implies&& y^2+26y+25 &< 0\\
                \implies&& (y+25)(y+1) &<0
            \end{alignat*}

            \begin{center}
                \begin{tikzpicture}
                        \draw[-latex] (-10,0) -- (0.0,0) node[right]{$y$};
                        \foreach \x in {-1.0,-8} \draw[shift={(\x,0)}] (0pt,3pt) -- (0pt,-3pt);
                        \draw[shift={(-1,-3pt)}] node[below]  {$-1$};
                        \draw[shift={(-8,-3pt)}] node[below]  {$-25$};

                        \draw[very thick, o-o, color=red] (-8.12, 0) -- (-0.88, 0);
                        \node[anchor=south, align=center] at (-9, 0) {$+$};
                        \node[anchor=south, align=center] at (-4.5, 0) {$-$};
                        \node[anchor=south, align=center] at (-0.5, 0) {$+$};
                \end{tikzpicture}
            \end{center}

            We thus see that $y$ cannot take on a value between -25 and -1.

            \boxt{
                $\{ y \in \mathbb{R} \colon -25 < y < -1 \}$
            }

        \part
            \begin{alignat*}{2}
                &&x^3+6x^2+3x-2 &= 0\\
                \implies&&\dfrac{x^3+6x^2+3x-2}{x+4} &= 0\\
                \implies&&\dfrac{x^3+4x^2}{x+4} + \dfrac{2x^2 + 3x-2}{x+4}&=0\\
                \implies&&x^2 + \dfrac{2x^2 + 3x-2}{x+4}&=0\\
                \implies&&C&=-x^2\\
            \end{alignat*}

            Plotting $y=-x^2$ on the same digram, we see that there are 3 intersections between $y=x^2$ and $C$. Hence, there are 3 distinct real roots to $x^3+6x^2+3x-2 = 0$.
