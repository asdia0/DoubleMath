    \problem{}
        The curve $C$ has equation $y = \dfrac{ax^2+bx}{x+2}$, where $x \neq -2$. It is given that $C$ has an asymptote $y = 1-2x$.

        \begin{enumerate}
            \item Show (do not verify) that $a = -2$ and $b = -3$.
            \item Using an algebraic method, find the set of values that $y$ can take.
            \item Sketch $C$, showing clearly the positions of any axial intercept(s), asymptote(s) and stationary point(s).
            \item Deduce that the equation $x^4 + 2x^3 + 2x^2 + 3x = 0 $ has exactly one real non-zero root.
        \end{enumerate}

    \solution
        \part
            \begin{equation*}
                \begin{aligned}
                    y &= \dfrac{ax^2+bx}{x+2} \\
                    &= \dfrac{(ax + b-2a)(x+2) - 2(b-2a)}{x+2}\\
                    &= ax+b-2a - \dfrac{2(b-2a)}{x+2}
                \end{aligned}
            \end{equation*}

            Since $C$ has an asymptote $y = 1-2x$, we have $a = -2$ and $b-2a = 1$, whence $b = -3$.

        \part
            \begin{alignat*}{2}
                &&y &= \dfrac{-2x^2+-3x}{x+2} \\
                \implies&&y(x+2) &= -2x^2 -3 x\\
                \implies&&2x^2 + (3+y)x + 2y &= 0
            \end{alignat*}

            For all values that $y$ can take on, there exists a solution to $2x^2 + (3+y)x + 2y = 0$. Hence, $\Delta \geq 0$.

            \begin{alignat*}{2}
                &&(3+y)^2 -4(2)(2y) &\geq 0\\
                \implies&&9 + 6y + y^2 - 16y &\geq 0\\
                \implies&&y^2-10y +9&\geq 0\\
                \implies&&(y-1)(y-9) &\geq 0
            \end{alignat*}

    \problem{}
        The curve $C$ is defined by the equation $y = \dfrac{x}{x^2-5x+4}$.

        \begin{enumerate}
            \item Write down the equations of the asymptotes.
            \item Sketch $C$, indicating clearly the axial intercept(s), asymptote(s) and turning point(s).
            \item Find the positive value $k$ such that the equation $\dfrac{x}{x^2-5x+4} = kx$ has exactly 2 distinct real roots.
        \end{enumerate}

    \solution
        \part
            As $x \rightarrow \pm \infty$, $y \rightarrow 0$. Hence, $y = 0$ is an asymptote. Observe that $x^2 - 5x + 4 = (x-1)(x-4)$. Hence, $x=1$ and $x=4$ are asymptotes.

            \boxt{
                Asymptotes: $y = 0$, $x=1$, $x=4$
            }

        \part
            \begin{center}
                \begin{tikzpicture}[trim axis left, trim axis right]
                    \begin{axis}[
                        domain = -2:7,
                        samples = 451,
                        axis y line=middle,
                        axis x line=middle,
                        xtick = \empty,
                        ytick = \empty,
                        xlabel = {$x$},
                        ylabel = {$y$},
                        ymax=10,
                        ymin=-10,
                        legend cell align={left},
                        legend pos=outer north east,
                        after end axis/.code={
                            \path (axis cs:0,0) 
                                node [anchor=north east] {$O$};
                            }
                        ]
                        \addplot[plotRed, unbounded coords = jump] {x/(x^2-5*x+4)};
            
                        \addlegendentry{$y = \tfrac{x}{x^2 -5x+4}$};
                        
                        \draw[dotted, very thick] (1, 10) -- (1, -10) node[anchor=south west] {$x=1$};

                        \draw[dotted, very thick] (4, 10) -- (4, -10) node[anchor=south west] {$x=4$};

                        \fill (2, -1) circle[radius=2.5 pt] node[anchor=north] {$(2,1)$};
                    \end{axis}
                \end{tikzpicture}
            \end{center}

        \part
            Note that $x = 0$ is always a root of $\dfrac{x}{x^2-5x+4} = kx$. We thus aim to find the value of $k$ such that $\dfrac{x}{x^2-5x+4} = kx$ has only one non-zero root. 

            We observe that if $k > 0$, $y=kx$ will intersect with $y = \dfrac{x}{x^2-5x+4}$ at least twice: before $x=1$ and after $x=4$. In order to have only one non-zero root, we must force the intersection point that comes before $x=1$ to be at the origin $(0, 0)$. Hence, $k$ is tangential to $C$ at $(0, 0)$, thus giving $k = \left.\der{C}{x}\right|_{x=0}$.

            \begin{equation*}
                \begin{aligned}
                    k &= \left.\der{C}{x}\right|_{x=0}\\
                    &= \left.\der{}{x}\dfrac{x}{x^2-5x+4}\right|_{x=0} \\
                    &= \left.\dfrac{3x^2-10x+4}{(x^2-5x+4)^2}\right|_{x=0} \\
                    &= \dfrac14
                \end{aligned}
            \end{equation*}

            \boxt{
                $k = \dfrac14$
            }
