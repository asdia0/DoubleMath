    \problem{}
        Calculate the area enclosed by the petals of the curve $r = \sin2\t$ where $r \geq 0$.

    \solution
        Note that $r \geq 0 \implies \sin2\t \geq 0 \implies r \in \left[0, \dfrac\pi2\right] \cup \left[\pi, \dfrac{3\pi}2\right]$.
        \begin{align*}
            \area &= 2 \cdot \dfrac12 \int_0^{\pi/2} \sin^2 2\t \d \t \usub{u &= 2\t\\\d u &= 2 \d \t\\\t = 0 &\implies u = 0\\\t = \dfrac\pi2 &\implies u = \pi}\\
            &= \dfrac12 \int_0^{\pi} \sin^2 u \d u\\
            &= \dfrac12 \int_0^{\pi} \dfrac{1 - \cos 2u}2 \d u\\
            &= \dfrac14 \int_0^{\pi} \bp{1 - \cos 2u} \d u\\
            &= \dfrac14 \evalint{u - \dfrac{\sin 2u}2}{0}{\pi}\\
            &= \dfrac\pi4
        \end{align*}

        \boxt{The area enclosed is $\dfrac\pi4$ units$^2$.}

    \problem{}
        The finite region $A$ is bounded by the curve $y = x^2$ and a minor arc of the circle $x^2 + y^2 = 12$.

        \begin{enumerate}
            \item Find the numerical value of the area of $A$, correct to 2 decimal places.
            \item Find the exact volume of the solid obtained when $A$ is rotated about the $x$-axis through $2\pi$ radians.
            \item Find the exact volume of the solid obtained when $A$ is rotated about the $y$-axis through $\pi$ radians.
        \end{enumerate}

    \solution
        \begin{center}
            \begin{tikzpicture}[trim axis left, trim axis right]
                \begin{axis}[
                    axis on top,
                    domain = -sqrt(12):sqrt(12),
                    samples = 101,
                    axis y line=middle,

                    \addplot[plotRed, name path=f1] {x^2};
        
                    \addlegendentry{$y = x^2$};
        
                    \addplot[plotBlue, name path=f2] {sqrt(12 - x^2)};
        
                    \addlegendentry{$x^2 + y^2 = 12$};

                    \addplot[color=black!20] fill between[of=f1 and f2,soft clip={domain=0:1.73}];

                    \addplot[color=black!20] fill between[of=f2 and f1,soft clip={domain=-1.73:0}];

                    \node[anchor=west] at (0, 1.73) {$A$};
                \end{axis}
            \end{tikzpicture}

        \part
            Consider the intersections between $y = x^2$ and $x^2 + y^2 = 12$.
            \begin{alignat*}{2}
                &&x^2 + y^2 &= 12\\
                \implies&&x^2 + \bp{x^2}^2 &= 12\\
                \implies&&x^4 + x^2 - 12 &= 0\\
                \implies&&(x^2 - 3)(x^2 + 4) &= 0\\
                \implies&&\bp{x - \sqrt3}\bp{x+\sqrt3}\bp{x^2 + 4} &= 0
            \end{alignat*}
            Hence, the two curves intersect at $x = -\sqrt3$ and $x = \sqrt3$. Note that $x^2 + 4 = 0$ has no solution since $x^2 + 4 > 0$. Also note that $x^2 + y^2 = 12 \implies y = \sqrt{12 - x^2}$ for $y > 0$.
            \begin{align*}
                \area &= 2\int_0^{\sqrt3} \bp{\sqrt{12-x^2} - x^2} \d x\\
                &= 8.02 \tosf{3}
            \end{align*}

            \boxt{$A$ has an area of 8.02 units$^2$.}

        \part
            Note that $x^2 + y^2 = 12 \implies y^2 = 12 - x^2$.
            \begin{align*}
                \volume &= 2\pi\int_0^{\sqrt3} \bp{\bp{12 - x^2} - \bp{x^2}^2} \d x\\
                &= 2\pi \evalint{12x - \dfrac13 x^3 - \dfrac15 x^5}{0}{\sqrt3}\\
                &= 2\pi \cdot \dfrac{46\sqrt{3}}5\\
                &= \dfrac{92\sqrt{3}\pi}5
            \end{align*}

            \boxt{The solid has a volume of $\dfrac{92\sqrt{3}\pi}5$ units$^3$.}

        \part
            Note that when the curves intersect at $x = \sqrt{3}$, we have $y = 3$. Furthermore, when $x = 0$, we have $y = \sqrt{12}$. Also note that $x^2 + y^2 = 12 \implies x^2 = 12 - y^2$.
            \begin{align*}
                \volume &= \pi\int_0^{3} y \d y + \pi \int_{3}^{\sqrt{12}} \bp{12 - y^2} \d y\\
                &= \pi\bp{16\sqrt{3} - \dfrac{45}2}
            \end{align*}

            \boxt{The solid has a volume of $\pi\bp{16\sqrt{3} - \dfrac{45}2}$ units$^3$.}

            \item Find $V$ in terms of $n$.
            \item State the limit of $V$ in exact form as $n \to \infty$.
        \end{enumerate}

    \solution
        \part
            \begin{align*}
                A &= \sum_{k=0}^{n-1} \dfrac{2^{k/n}}n\\
                &= \dfrac1n \sum_{k=0}^{n-1} \bp{2^{1/n}}^k\\
                &= \dfrac1n \cdot \dfrac{1 - \bp{2^{1/n}}^{n-1+1}}{1 - 2^{1/n}}\\
                &= \dfrac1n \cdot \dfrac{1 -2}{1 - 2^{1/n}}\\
                &= \dfrac1n \cdot \dfrac{1}{2^{1/n} - 1}
            \end{align*}

        \part
            \begin{align*}
                \lim_{n \to \infty} A &= \lim_{n \to \infty} \dfrac1n \cdot \dfrac{1}{2^{1/n} - 1}\\
                &= \lim_{n \to \infty} \dfrac{1/n}{2^{1/n} - 1}\\
                &= \lim_{m \to 0} \dfrac{m}{2^m - 1}\\
                &= \lim_{m \to 0} \dfrac{1}{\ln 2 \cdot 2^m}\\
                &= \dfrac1{\ln 2}
            \end{align*}

            \boxt{$\lim\limits_{n \to \infty} A = \dfrac1{\ln 2}$}

        \part
            \begin{align*}
                V &= \pi \sum_{k=0}^{n-1} \bp{2^{k/n}}^2 \cdot \dfrac1n\\
                &= \dfrac\pi{n} \sum_{k=0}^{n-1} \bp{2^{2/n}}^k\\
                &= \dfrac\pi{n} \cdot \dfrac{1 - \bp{2^{2/n}}^{n-1+1}}{1 - 2^{2/n}}\\
                &= \dfrac\pi{n} \cdot \dfrac{1 - 4}{1 - 2^{2/n}}\\
                &= \dfrac\pi{n} \cdot \dfrac{3}{2^{2/n} - 1}\\
                &= \dfrac\pi{n} \cdot \dfrac{3}{4^{1/n} - 1}\\
                &= \dfrac{3\pi}{n\bp{4^{1/n} - 1}}
            \end{align*}

            \boxt{$V = \dfrac{3\pi}{n\bp{4^{1/n} - 1}}$}

        \part
            \begin{align*}
                \lim_{n \to \infty} V &= \lim_{n \to \infty} \dfrac{3\pi}{n\bp{4^{1/n} - 1}}\\
                &= 3\pi \lim_{n \to \infty} \dfrac{1}{n\bp{4^{1/n} - 1}}\\
                &= 3\pi \lim_{n \to \infty} \dfrac{1/n}{4^{1/n} - 1}\\
                &= 3\pi \lim_{m \to 0} \dfrac{m}{4^m - 1}\\
                &= 3\pi \lim_{m \to 0} \dfrac{1}{\ln 4 \cdot 4^m}\\
                &= 3\pi \cdot \dfrac1{\ln 4}\\
                &= \dfrac{3\pi}{2\ln2}
            \end{align*}

            \boxt{$\lim\limits_{n \to \infty} V = \dfrac{3\pi}{2\ln2}$}

        \begin{center}
            \begin{tikzpicture}[trim axis left, trim axis right]
                \begin{axis}[
                    domain = 0:13,
                    samples = 101,
                    axis y line=middle,
                    axis x line=middle,
                    xtick = \empty,
                    ytick = \empty,
                    xlabel = {$t$},
                    ylabel = {$v$},
                    legend cell align={left},
                    legend pos=outer north east,
                    after end axis/.code={
                        \path (axis cs:0,0) 
                            node [anchor=north east] {$O$};
                        }
                    ]
                    \addplot[plotRed] {x * (x-10)};
                \end{axis}
            \end{tikzpicture}
        \end{center}
        The diagram shows the velocity-time graph of a particle moving in a straight line. The equation of the curve shown is $v = t(t - 10)$ where $t$ seconds is the time and $v$ ms$^{-1}$ is the velocity. The particle starts at a point $A$ on the line when $t = 0$.

        Calculate
        \begin{enumerate}
            \item the distance travelled by the particle before coming to instantaneous rest, and
            \item the time at which the particle returns to $A$.
        \end{enumerate}

    \solution
        \part
            For instantaneous rest, $v = 0$. Hence, $t(t-10) = 0$, whence $t = 10$. Note that we reject $t = 0$ since $t > 0$.
            \begin{align*}
                \text{Distance travelled} &= -\int_0^{10} v \d t\\
                &= -\int_0^{10} t(t-10) \d t\\
                &= -\int_0^{10} \bp{t^2 - 10t} \d t\\
                &= -\evalint{\dfrac{t^3}3 - \dfrac{10t^2}2}{0}{10}\\
                &= \dfrac{500}3
            \end{align*}
            
            \boxt{The particle travelled $\dfrac{500}3$ m before coming to instantaneous rest.}

        \part
            When the particle returns to $A$, $s = 0$. Let the time at which the particle returns to $A$ be $t_0$.
            \begin{alignat*}{2}
                &&\int_0^{t_0} v \d t &= 0\\
                \implies&&\int_0^{t_0} t(t-10) \d t &= 0\\
                \implies&&\evalint{\dfrac{t_0^3}3 - \dfrac{10t_0^2}2}{0}{t_0} &= 0\\
                \implies&&\dfrac13 t_0^3 - 5t_0^2 &= 0\\
                \implies&&\dfrac13t_0^2 \bp{t_0 - 15} &= 0
            \end{alignat*}

            Thus, $t_0 = 15$. Note that we reject $t_0 = 0$ since $t_0 > 0$.

            \boxt{It takes the particle 15 seconds to return to $A$.}
